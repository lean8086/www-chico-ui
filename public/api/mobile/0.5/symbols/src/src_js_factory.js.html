1 /** 
  2 * Creational patterns to create UI Components
  3 * @name factory
  4 * @class Factory
  5 * @static
  6 * @param o Configuration Object
  7 * @returns {Object}
  8 * @memberOf ch
  9 */
 10 ch.factory = function (klass) {
 11 	// Sets the klass name
 12 	var name = klass.toLowerCase(),
 13 
 14 		// Creates a new instance
 15 		create = function (e, conf) {
 16 
 17 			var context = {
 18 				"type": name,
 19 				"el": e,
 20 				"$el": $(e),
 21 				"uid": uid += 1 // Global instantiation index
 22 			};
 23 
 24 			switch (typeof conf) {
 25 			// If argument is a number, join with the conf
 26 			case "number":
 27 				var num = conf;
 28 				conf = {};
 29 				conf.value = num;
 30 
 31 				// Could come a messages as a second argument
 32 				if (arguments[1]) {
 33 					conf.msg = arguments[1];
 34 				}
 35 			break;
 36 
 37 			// This could be a message
 38 			case "string":
 39 				var msg = conf;
 40 				conf = {};
 41 				conf.msg = msg;
 42 			break;
 43 
 44 			// This is a condition for custom validation
 45 			case "function":
 46 				var func = conf;
 47 				conf = {};
 48 				conf.lambda = func;
 49 				
 50 				// Could come a messages as a second argument
 51 				if (arguments[1]) {
 52 					conf.msg = arguments[1];
 53 				}
 54 			break;
 55 			};
 56 
 57 			var created = ch[klass].call(context, conf);
 58 			created = (hasOwn(created, "public")) ? created["public"] : created;
 59 
 60 			if (created.type) {
 61 				var type = created.type;
 62 				// If component don't exists in the instances map create an empty array
 63 				if (!ch.instances[type]) { 
 64 					ch.instances[type] = [];
 65 				}
 66 				ch.instances[type].push(created);
 67 			}
 68 
 69 			// Avoid mapping objects that already exists
 70 			if (created.exists) {
 71 				// Return the inner object
 72 				created = created.object;
 73 			}
 74 
 75 			return created;
 76 		};
 77 
 78 	
 79 	// Add class constructor to ch
 80 	$.fn[name] = function (conf) {
 81 		// Collection of instances
 82 		var output = [],
 83 			instantiated = [];
 84 
 85 		$.each(this, function (i, e) {
 86 			if (hasOwn(ch.instances, name)) {
 87 				$.each(ch.instances[name], function (i, instance) {
 88 					if (e === instance.el) {
 89 						instantiated.push(instance);
 90 					}
 91 				});
 92 			}
 93 
 94 			output.push(((instantiated.length > 1) ? instantiated : instantiated[0]) ||Â create(e, conf));
 95 		});
 96 
 97 		// Return the instances or collection of instances
 98 		return (output.length > 1) ? output : output[0];
 99 	};
100 
101 	return this;
102 };
