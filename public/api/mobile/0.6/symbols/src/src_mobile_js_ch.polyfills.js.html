 20 if (!("classList" in document.documentElement) && Object.defineProperty && typeof HTMLElement !== 'undefined') {
 21     Object.defineProperty(HTMLElement.prototype, 'classList', {
 22         get: function() {
 23             var self = this;
 24             function update(fn) {
 25                 return function(value) {
 26                     var classes = self.className.split(/\s+/),
 27                         index = classes.indexOf(value);
 28 
 29                     fn(classes, index, value);
 30                     self.className = classes.join(" ");
 31                 }
 32             }
 33 
 34             var ret = {                    
 35                 add: update(function(classes, index, value) {
 36                     ~index || classes.push(value);
 37                 }),
 38 
 39                 remove: update(function(classes, index) {
 40                     ~index && classes.splice(index, 1);
 41                 }),
 42 
 43                 toggle: update(function(classes, index, value) {
 44                     ~index ? classes.splice(index, 1) : classes.push(value);
 45                 }),
 46 
 47                 contains: function(value) {
 48                     return !!~self.className.split(/\s+/).indexOf(value);
 49                 },
 50 
 51                 item: function(i) {
 52                     return self.className.split(/\s+/)[i] || null;
 53                 }
 54             };
 55             
 56             Object.defineProperty(ret, 'length', {
 57                 get: function() {
 58                     return self.className.split(/\s+/).length;
 59                 }
 60             });
 61 
 62             return ret;
 63         }
 64     });
