1 /**
  2 * Viewport is a reference to position and size of the visible area of browser.
  3 * @abstract
  4 * @name Viewport
  5 * @class Viewport
  6 * @standalone
  7 * @memberOf ch
  8 */
  9 ch.viewport = {
 10 
 11 	/**
 12 	* Width of the visible area.
 13 	* @public
 14 	* @name ch.Viewport#width
 15 	* @type Number
 16 	*/
 17 	"width": ch.utils.window.width(),
 18 
 19 	/**
 20 	* Height of the visible area.
 21 	* @public
 22 	* @name ch.Viewport#height
 23 	* @type Number
 24 	*/
 25 	"height": ch.utils.window.height(),
 26 
 27 	/**
 28 	* Left offset of the visible area.
 29 	* @public
 30 	* @name ch.Viewport#left
 31 	* @type Number
 32 	*/
 33 	"left": ch.utils.window.scrollLeft(),
 34 
 35 	/**
 36 	* Top offset of the visible area.
 37 	* @public
 38 	* @name ch.Viewport#top
 39 	* @type Number
 40 	*/
 41 	"top": ch.utils.window.scrollTop(),
 42 
 43 	/**
 44 	* Right offset of the visible area.
 45 	* @public
 46 	* @name ch.Viewport#right
 47 	* @type Number
 48 	*/
 49 	"right": ch.utils.window.scrollLeft() + ch.utils.window.width(),
 50 
 51 	/**
 52 	* Bottom offset of the visible area.
 53 	* @public
 54 	* @name ch.Viewport#bottom
 55 	* @type Number
 56 	*/
 57 	"bottom": ch.utils.window.scrollTop() + ch.utils.window.height(),
 58 
 59 	/**
 60 	* Element representing the visible area.
 61 	* @public
 62 	* @name ch.Viewport#element
 63 	* @type Object
 64 	*/
 65 	"element": ch.utils.window,
 66 
 67 	/**
 68 	* Updates width and height of the visible area and updates ch.viewport.width and ch.viewport.height
 69 	* @public
 70 	* @function
 71 	* @name ch.Viewport#getSize
 72 	* @returns Object
 73 	*/
 74 	"getSize": function () {
 75 
 76 		return {
 77 			"width": this.width = ch.utils.window.width(),
 78 			"height": this.height = ch.utils.window.height()
 79 		};
 80 
 81 	},
 82 
 83 	/**
 84 	* Updates left, top, right and bottom coordinates of the visible area, relative to the window.
 85 	* @public
 86 	* @function
 87 	* @name ch.Viewport#getPosition
 88 	* @returns Object
 89 	*/
 90 	"getPosition": function () {
 91 
 92 		var size = this.getSize();
 93 
 94 		return {
 95 			"left": 0,
 96 			"top": 0,
 97 			"right": size.width,
 98 			"bottom": size.height,
 99 			// Size is for use as context on Positioner
100 			// (see getCoordinates method on Positioner)
101 			"width": size.width,
102 			"height": size.height
103 		};
104 		
105 	},
106 	
107 	/**
108 	* Updates left, top, right and bottom coordinates of the visible area, relative to the document.
109 	* @public
110 	* @function
111 	* @name ch.Viewport#getOffset
112 	* @returns Object
113 	*/
114 	"getOffset": function () {
115 
116 		var position = this.getPosition(),
117 			scrollLeft = ch.utils.window.scrollLeft(),
118 			scrollTop = ch.utils.window.scrollTop();
119 
120 		return {
121 			"left": this.left = scrollLeft,
122 			"top": this.top = scrollTop,
123 			"right": this.right = scrollLeft + position.right,
124 			"bottom": this.bottom = scrollTop + position.bottom
125 		};
126 		
127 	}
128 };
