 20 * });
 21 * @exampleDescription Create a new condition object with expr.
 22 * @example
 23 * var widget = ch.condition({
 24 *     "name": "maxLength",
 25 *     "patt": function(a,b) { return a.length <= b },
 26 *     "message": "Some message here!",
 27 *     "value": 4
 28 * });
 29 * @exampleDescription Create a new condition object with func.
 30 * @example
 31 * var widget = ch.condition({
 32 *     "name": "custom",
 33 *     "patt": function (value) {
 34 *         if (value === "ChicoUI") {
 35 *

 53 	function Condition(condition) {
 54 
 55 		if (condition === undefined) {
 56 			throw new window.Error('"ch.Condition(condition)": Expected condition be defined and be an object.');
 57 		}
 58 
 59 		if ((condition.name === undefined) || (typeof condition.name !== 'string')) {
 60 			throw new window.Error('"ch.Condition({ \'name\': \'custom\' })": Expected property name be defined and be a string.');
 61 		}
 62 
 63 		if ((condition.name === 'custom') && (typeof condition.fn !== 'function')) {
 64 			throw new window.Error('"ch.Condition({ \'name\': \'custom\', \'fn\': function(){} })": Expected property "fn" be defined as a function with "custom" condition name.');
 65 		}
 66 
 67 		if ((condition.name === 'min') && ((condition.num === undefined) || (typeof condition.num !== 'number'))) {
 68 			throw new window.Error('"ch.Condition({ \'name\': \'min\', \'num\': 10 })": Expected property "num" be defined as a number with "min" condition name.');
 69 		}
 70 
 71 		if ((condition.name === 'max') && ((condition.num === undefined) || (typeof condition.num !== 'number'))) {
 72 			throw new window.Error('"ch.Condition({ \'name\': \'max\', \'num\': 10 })": Expected property "num" be defined as a number with "max" condition name.');
 73 		}
 74 
 75 		if ((condition.name === 'minLength') && ((condition.num === undefined) || (typeof condition.num !== 'number'))) {
 76 			throw new window.Error('"ch.Condition({ \'name\': \'minLength\', \'num\': 10 })": Expected property "num" be defined as a number with "minLength" condition name.');
 77 		}
 78 
 79 		if ((condition.name === 'maxLength') && ((condition.num === undefined) || (typeof condition.num !== 'number'))) {
 80 			throw new window.Error('"ch.Condition({ \'name\': \'maxLength\', \'num\': 10 })": Expected property "num" be defined as a number with "maxLength" condition name.');
 81 		}

 88 	*/
 89 		var that = this;

 95 		var conditions = {
 96 			'string': {
 97 				// the following regular expression has the utf code for the lating characters
 98 				// the ranges are A,EI,O,U,a,ei,o,u,ç,Ç please for reference see http://www.fileformat.info/info/charset/UTF-8/list.htm
 99 				'patt': /^([a-zA-Z\u00C0-\u00C4\u00C8-\u00CF\u00D2-\u00D6\u00D9-\u00DC\u00E0-\u00E4\u00E8-\u00EF\u00F2-\u00F6\u00E9-\u00FC\u00C7\u00E7\s]*)$/,
100 				'message': 'Use only letters.'
101 			},
102 			'email': {
103 				'patt': /^[a-zA-Z0-9.!#$%&’*+/=?^_`{|}~-]+@[a-zA-Z0-9-]+(?:\.[a-zA-Z0-9-]+)*$/,
104 				'message': 'Use a valid e-mail such as name@example.com.'
105 			},
106 			'url': {
107 				'patt': /^((https?|ftp|file):\/\/|((www|ftp)\.)|(\/|.*\/)*)[a-z0-9-]+((\.|\/)[a-z0-9-]+)+([/?].*)?$/,
108 				'message': 'It must be a valid URL.'
109 			},
110 			'minLength': {
111 				'expr': function(a,b) { return a.length >= b },
112 				'message': 'Enter at least {#num#} characters.'
113 			},
114 			'maxLength': {
115 				'expr': function(a,b) { return a.length <= b },
116 				'message': 'The maximum amount of characters is {#num#}.'
117 			},
118 			'number': {
119 				'patt': /^(-?[0-9\s]+)$/,
120 				'message': 'Use only numbers.'
121 			},
122 			'max': {
123 				'expr': function(a, b) { return a <= b },
124 				'message': 'The amount must be smaller than {#num#}.'
125 			},
126 			'min': {
127 				'expr': function(a, b) { return a >= b },
128 				'message': 'The amount must be higher than {#num#}.'
129 			},
130 			'price': {
131 				'patt': /^(\d+)[.,]?(\d?\d?)$/,
132 				'message': 'Use a valid price such as (1,00).'
133 			},
134 			'required': {
135 				'expr': function(e) {
136 
137 					if (e === undefined) {
138 						throw new window.Error('"instance.test(HTMLElement)": The "HTMLElement" parameter must be defined and be HTMLElement object.');
139 					}
140 
141 					var $e = $(e);
142 
143 					var tag = $e.hasClass("ch-form-options") ? "OPTIONS" : e.tagName;
144 					switch (tag) {
145 						case 'OPTIONS':
146 							return $e.find('input:checked').length !== 0;
147 						break;
148 
149 						case 'SELECT':
150 							var val = $e.val();
151 							return (val != "-1" && val != "");
152 						break;
153 
154 						case 'INPUT':
155 						case 'TEXTAREA':
156 							return $.trim($e.val()).length !== 0;
157 						break;
158 					};
159 				},
160 				'message': 'Fill in this information.'
161 			},
162 			'custom': {
163 				// I don't have pre-conditions, comes within conf.fn argument
164 				'message': 'Error'
165 			}
166 		};
167 
168 		/**
169 		* Flag that let you know if the condition is enabled or not.
170 		* @private
171 		* @name ch.Condition-enabled
172 		* @type boolean
173 		*/
174 		var	enabled = true,
175 
176 			test = function (value) {
177 
178 				if (value === undefined) {
179 					throw new window.Error('"instance.test(value)": The "value" parameter must be defined.');
180 				}
181 
182 				if (!enabled) {
183 					return true;
184 				}
185 
186 				if (condition.patt){
187 					return condition.patt.test(value);
188 				}
189 
190 				if (condition.expr){
191 					return condition.expr(value, condition.num);
192 				}
193 
194 				if (condition.fn){
195 					// Call validation function with 'this' as scope.
196 					return condition.fn.call(this, value);
197 				}
198 
199 			},
200 
201 			enable = function() {
202 				enabled = true;
203 
204 				return condition;
205 			},
206 
207 			disable = function() {
208 				enabled = false;
209 
210 				return condition;
211 			};

234

238 		* @function
239 		* @name ch.Condition#test
240 		* @returns boolean
241 		*/
242 
243 		/**
244 		* Turn on condition.
245 		* @public
246 		* @name ch.Condition#enable
247 		* @function

249 		*/
250

253 		* @public

255 		* @function

258 
259 		if (conditions[condition.name] === undefined) {
260 			throw new window.Error('"ch.Condition({ \'name\': \''+condition.name+'\' })": "'+condition.name+'" Condition is not defined. Use "custom" to define your own condition.');
261 		}
262 
263 		// replaces the condition default message in the following conditions max, min, minLenght, maxLenght
264 		if(!condition.message && (condition.name === 'min' || condition.name === 'max' || condition.name === 'minLength' || condition.name === 'maxLength')){
265 
266 			conditions[condition.name].message = conditions[condition.name].message.replace('{#num#}', condition.num);
267 
268 		}
269 
270 		// condition override the default conditions configuration
271 		condition = $.extend({}, conditions[condition.name], condition, {
272 			test: test,
273 			enable: enable,
274 			disable: disable
275 		});
276 
277 		return condition;
278 
279 	};
