1 /**
  2 * Validate strings.
  3 * @name String 
  4 * @class String
  5 * @interface
  6 * @augments ch.Uiobject
  7 * @requires ch.Watcher
  8 * @memberOf ch
  9 * @param {String} [message] Validation message
 10 * @returns itself
 11 * @see ch.Watcher
 12 * @see ch.Required
 13 * @see ch.Custom
 14 * @see ch.Number
 15 * @see ch.Validator
 16 * @see ch.Condition
 17 * @example
 18 * // Create a string validation
 19 * $("input").string("This field must be a string.");
 20 */
 21 ch.extend("watcher").as("string", function (conf) {
 22 
 23 	conf.condition = {
 24 		name: "string",
 25 		// the following regular expression has the utf code for the lating characters
 26 		// the ranges are A,EI,O,U,a,ei,o,u,รง,ร please for reference see http://www.fileformat.info/info/charset/UTF-8/list.htm
 27 		patt: /^([a-zA-Z\u00C0-\u00C4\u00C8-\u00CF\u00D2-\u00D6\u00D9-\u00DC\u00E0-\u00E4\u00E8-\u00EF\u00F2-\u00F6\u00E9-\u00FC\u00C7\u00E7\s]*)$/,
 28 		message: conf.msg || conf.message
 29 	};
 30 
 31 	return conf;
 32 
 33 });

 35 /**
 36 * Validate email sintaxis.
 37 * @name Email
 38 * @class Email
 39 * @interface
 40 * @augments ch.Uiobject
 41 * @requires ch.Watcher
 42 * @memberOf ch
 43 * @param {String} [message] Validation message
 44 * @returns itself
 45 * @see ch.Watcher
 46 * @see ch.Required
 47 * @see ch.Custom
 48 * @see ch.Number
 49 * @see ch.Validator
 50 * @see ch.Condition
 51 * @example
 52 * // Create a email validation
 53 * $("input").email("This field must be a valid email.");
 54 */
 55 ch.extend("watcher").as("email", function (conf) {
 56 
 57 	conf.condition = {
 58 		name: "email",
 59 		patt: /^(("[\w-\s]+")|([\w-]+(?:\.[\w-]+)*)|("[\w-\s]+")([\w-]+(?:\.[\w-]+)*))(@((?:[\w-]+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z]{2})?)$)|(@\[?((25[0-5]\.|2[0-4][0-9]\.|1[0-9]{2}\.|[0-9]{1,2}\.))((25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\.){2}(25[0-5]|2[0-4][0-9]|1[0-9]{2}|[0-9]{1,2})\]?$)/,
 60 		message: conf.msg || conf.message
 61 	};
 62 
 63 	return conf;
 64 
 65 });

 88 
 89 	conf.condition = {
 90 		name: "url",
 91 		patt: /^((https?|ftp|file):\/\/|((www|ftp)\.)|(\/|.*\/)*)[a-z0-9-]+((\.|\/)[a-z0-9-]+)+([/?].*)?$/,
 92 		message: conf.msg || conf.message
 93 	};
 94 
 95 	return conf;

100 /**
101 * Validate a minimun amount of characters.
102 * @name MinLength
103 * @class MinLength
104 * @interface
105 * @augments ch.Uiobject
106 * @requires ch.Watcher
107 * @memberOf ch
108 * @param {Number} value Minimun number value.
109 * @param {String} [message] Validation message
110 * @returns itself
111 * @see ch.Watcher
112 * @see ch.Required
113 * @see ch.Custom
114 * @see ch.Number
115 * @see ch.Validator
116 * @see ch.Condition
117 * @example
118 * // Create a minLength validation
119 * $("input").minLength(10, "At least 10 characters..");
120 */
121 ch.extend("watcher").as("minLength", function (conf) {
122 
123 	conf.condition = {
124 		name: "minLength",
125 		expr: function(a,b) { return a.length >= b },
126 		message: conf.msg || conf.message,
127 		value: conf.value
128 	};
129 
130 	return conf;
131 
132 });
