1 /**
  2 * Forms is a Controller of DOM's HTMLFormElement.
  3 * @name Form
  4 * @class Form
  5 * @augments ch.Controllers
  6 * @memberOf ch
  7 * @param {Object} [conf] Object with configuration properties.
  8 * @param {Object} [conf.messages]
  9 * @see ch.Validation
 10 * @see ch.Controllers
 11 * @returns itself
 12 * @factorized
 13 * @exampleDescription Create a new Form.
 14 * @example
 15 * var widget = $(".example").form();
 16 * @exampleDescription Create a new Form with some messages that will be use the validation engine.
 17 * @example
 18 * var widget = $(".example").form({
 19 * 	"messages": {
 20 * 		"required": "Error message for all required fields.",
 21 * 		"email": "Show this message on email format error."
 22 * 	}
 23 * });
 24 */
 25 
 26 ch.form = function(conf) {
 27 
 28 /**
 29 * Validation
 30 */
 31 	// Are there action and submit type?
 32 	if ( this.$element.find(":submit").length == 0 || this.$element.attr("action") == "" ){
 33 		alert("Form fatal error: The <input type=submit> is missing, or need to define a action attribute on the form tag.");
 34 		return;
 35 	};
 36 
 37 	// Is there form in map instances?
 38 	if ( ch.utils.hasOwn(ch.instances, "form") && ch.instances.form.length > 0 ){
 39 		for(var i = 0, j = ch.instances.form.length; i < j; i+=1){
 40 			if(ch.instances.form[i].element === this.element){
 41 				return {
 42 					exists: true,
 43 					object: ch.instances.form[i]
 44 				};
 45 			}
 46 		};
 47 	}
 48 
 49 	/**
 50 	* Reference to a internal component instance, saves all the information and configuration properties.
 51 	* @private
 52 	* @name ch.Form#that
 53 	* @type object
 54 	*/
 55 	var that = this;
 56 
 57 	conf = ch.clon(conf);
 58 	// Disable HTML5 browser-native validations
 59 	that.$element.attr("novalidate", "novalidate");
 60 	// Grab submit button
 61 	that.$submit = that.$element.find("input:submit");
 62 
 63 	that.conf = conf;
 64 
 65 /**
 66 *  Inheritance
 67 */
 68 
 69 	that = ch.controllers.call(that);
 70 	that.parent = ch.clon(that);
 71 	
 72 /**
 73 *  Private Members
 74 */
 75 
 76 	/**
 77 	* A Boolean property that indicates is exists errors in the form.
 78 	* @private
 79 	* @name ch.Form#status
 80 	* @type boolean
 81 	*/
 82 	var status = true;
 83 
 84 	/**
 85 	* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
 86 	*/
 87 	var validate = function(event){
 88 
 89 		/**
 90 		* Fired before the validations engine start. 
 91 		* @name ch.Form#beforeValidate
 92 		* @event
 93 		* @public
 94 		* @exampleDescription 
 95 		* @example
 96 		* widget.on("beforeValidate",function () {
 97 		*	sowidget.action();
 98 		* });
 99 		*/
100 		that.callbacks("beforeValidate");
101 		// new callback
102 		that.trigger("beforeValidate");
103 
104 		// Status OK (with previous error)
105 		if ( !status ) {
106 			status = true;
107 		};
108 
109 		var i = 0, j = that.children.length, toFocus, childrenError = [];
110 
111 		// Shoot validations
112 		for (i; i < j; i+=1) {
113 			var child = that.children[i];
114 
115 			// Validate
116 			// Save children with errors
117 			if ( child.hasError() ) {
118 				childrenError.push(child);
119 			}
120 		};
121 
122 		// Is there's an error
123 		if (childrenError.length > 0) {
124 			status = false;
125 			// Issue UI-332: On validation must focus the first field with errors.
126 			// Doc: http://wiki.ml.com/display/ux/Mensajes+de+error
127 			if (childrenError[0].element.tagName === "DIV") {
128 				$(childrenError[0].element).find("input:first").focus();
129 			} else if (childrenError[0].element.type !== "hidden") {
130 				childrenError[0].element.focus();
131 			}
132 		} else {
133 			status = true;
134 		}
135 
136 		/**
137 		* Fired when the form validates. 
138 		* @name ch.Form#validate
139 		* @event
140 		* @public
141 		* @exampleDescription 
142 		* @example
143 		* widget.on("validate",function () {
144 		*	sowidget.action();
145 		* });
146 		*/
147 
148 		/**
149 		* Fired when the form fall on a error. 
150 		* @name ch.Form#error
151 		* @event
152 		* @public
153 		* @exampleDescription 
154 		* @example
155 		* widget.on("error",function () {
156 		*	sowidget.action();
157 		* });
158 		*/
159 		if (status) {
160 			that.callbacks("onValidate");
161 			// new callback
162 			that.trigger("validate");
163 		} else {
164 			that.callbacks("onError");
165 			// new callback
166 			that.trigger("error");
167 		}
168 
169 		/**
170 		* Fired when the validations end.
171 		* @name ch.Form#afterValidate
172 		* @event
173 		* @public
174 		* @exampleDescription 
175 		* @example
176 		* widget.on("afterValidate",function () {
177 		*	sowidget.action();
178 		* });
179 		*/
180 		that.callbacks("afterValidate");
181 		// new callback
182 		that.trigger("afterValidate");
183 
184 		return that;
185 	};
186 
187 	/**
188 	* This methods triggers the 'beforSubmit' callback, then will execute validate() method,
189 	* and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
190 	*/
191 	var submit = function(event) {
192 
193 		/**
194 		* Fired before the form's submition.
195 		* @name ch.Form#beforeSubmit
196 		* @event
197 		* @public
198 		* @exampleDescription 
199 		* @example
200 		* widget.on("beforeSubmit",function () {
201 		*	sowidget.action();
202 		* });
203 		*/
204 		that.callbacks("beforeSubmit");
205 		// new callback
206 		that.trigger("beforeSubmit");
207 
208 		// Execute all validations
209 		validate(event);
210 
211 		// If an error occurs prevent default actions
212 		if (!status) {
213 			that.prevent(event);
214 	        if (event) {
215 	            event.stopImmediatePropagation();
216 	        }
217 		}
218 
219 		// OLD CALLBACK SYSTEM!
220 		// Is there's no error but there's a onSubmit callback
221 		if ( status && ch.utils.hasOwn(conf, "onSubmit")) {
222 			// Avoid default actions
223 			that.prevent(event);
224 			// To execute defined onSubmit callback
225 			that.callbacks("onSubmit");
226 		}
227 
228 		/**
229 		* Fired when submits the form.
230 		* @name ch.Form#submit
231 		* @event
232 		* @public
233 		* @exampleDescription 
234 		* @example
235 		* widget.on("afterSubmit",function () {
236 		*	sowidget.action();
237 		* });
238 		*/
239 		// * New callback system *
240 		// Check inside $.data if there's a handler for ch-submit event
241 		// if something found there, avoid submit.
242 
243 		var formEvents = $(that["public"]).data("events");
244 		var isSubmitEventDefined = (formEvents && ch.utils.hasOwn(formEvents, "ch-submit"));
245 
246 		if (status && isSubmitEventDefined){
247 			// Avoid default actions
248 			that.prevent(event);
249 			// new callback
250 			that.trigger("submit");
251 		};
252 
253 		/**
254 		* Fired after the form's submition.
255 		* @name ch.Form#afterSubmit
256 		* @event
257 		* @public
258 		* @exampleDescription 
259 		* @example
260 		* widget.on("afterSubmit",function () {
261 		*	this.reset();
262 		* });
263 		*/
264 		that.callbacks("afterSubmit");
265 		// new callback
266 		that.trigger("afterSubmit");
267 
268 		// Return that to chain methods
269 		return that;
270 	};
271 
272 	/**
273 	* Use this method to clear al validations.
274 	*/
275 	var clear = function(){
276 
277 		var i = 0, j = that.children.length;
278 		for(i; i < j; i += 1) {
279 			that.children[i].clear();
280 		}
281 
282 		status = true;
283 
284 		/**
285 		* Fired when clean the form's data.
286 		* @name ch.Form#clear
287 		* @event
288 		* @public
289 		* @exampleDescription 
290 		* @example
291 		* widget.on("clear",function () {
292 		*	this.reset();
293 		* });
294 		*/
295 		that.callbacks("onClear");
296 		// new callback
297 		that.trigger("clear");
298 
299 		return that;
300 	};
301 
302 	/**
303 	* Use this method to reset the form's input elements.
304 	*/
305 	var reset = function(event){
306 		clear();
307 		that.element.reset(); // Reset html form native
308 
309 		/**
310 		* Fired when resets the form.
311 		* @name ch.Form#reset
312 		* @event
313 		* @public
314 		* @exampleDescription 
315 		* @example
316 		* widget.on("reset",function () {
317 		*	sowidget.action();
318 		* });
319 		*/
320 		that.callbacks("onReset");
321 		// new callback
322 		that.trigger("reset");
323 
324 		return that;
325 	};

361 	/**
362 	* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
363 	* @public
364 	* @function
365 	* @name ch.Form#validate
366 	* @returns itself
367 	*/
368 	that["public"].validate = function() {
369 		validate();
370 
371 		return that["public"];
372 	};
373 
374 	/**
375 	* This methods triggers the 'beforSubmit' callback, then will execute validate() method, and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
376 	* @public
377 	* @function
378 	* @name ch.Form#submit
379 	* @returns itself
380 	*/
381 	that["public"].submit = function() {
382 		submit();
383 
384 		return that["public"];
385 	};
386 
387 	/**
388 	* Return the status value.
389 	* @public
390 	* @function
391 	* @name ch.Form#getStatus
392 	* @returns itself
393 	*/
394 	that["public"].getStatus = function(){
395 		return status;
396 	};
397 
398 	/**
399 	* Use this method to clear al validations.
400 	* @public
401 	* @function
402 	* @function
403 	* @name ch.Form#clear
404 	* @returns itself
405 	*/
406 	that["public"].clear = function() {
407 		clear();
408 
409 		return that["public"];
410 	};
411 
412 	/**
413 	* Use this method to clear al validations.
414 	* @public
415 	* @function
416 	* @name ch.Form#reset
417 	* @returns itself
418 	*/
419 	that["public"].reset = function() {
420 		reset();
421 
422 		return that["public"];
423 	};
424 
425 /**
426 *  Default event delegation
427 */
428 
429 	// patch exists because the components need a trigger
430 	if (ch.utils.hasOwn(conf, "onSubmit")) {
431 		that.$element.bind('submit', function(event){ that.prevent(event); });
432 		// Delete all click handlers asociated to submit button >NATAN: Why?
433 			//Because if you want to do something on submit, you need that the trigger (submit button)
434 			//don't have events associates. You can add funcionality on onSubmit callback
435 		that.$element.find(":submit").unbind('click');
436 	};
437 
438 	// Bind the submit
439 	that.$element.bind("submit", function(event){ submit(event) });
440 
441 	// Bind the reset
442 	that.$element.find(":reset, .resetForm").bind("click", function(event){ reset(event); });
443 
444 	/**
445 	* Triggers when the component is ready to use (Since 0.8.0).
446 	* @name ch.Form#ready
447 	* @event
448 	* @public
449 	* @since 0.8.0
450 	* @exampleDescription Following the first example, using <code>widget</code> as form's instance controller:
451 	* @example
452 	* widget.on("ready",function () {
453 	*	this.reset();
454 	* });
455 	*/
456 	setTimeout(function(){ that.trigger("ready")}, 50);
457 
458 	return that;
459 };
