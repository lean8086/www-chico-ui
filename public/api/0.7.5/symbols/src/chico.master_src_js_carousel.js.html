 51 	/**
 52 	* Does what is necessary to make ready the component structure.
 53 	* @private
 54 	* @name ch.Carousel#createLayout
 55 	* @function
 56 	*/
 57 	var createLayout = function () {
 58 		
 59 		// Add class to component to support old HTML snippet
 60 		that.$element.addClass("ch-carousel");
 61 		
 62 		// Calculate extra width for content
 63 		extraWidth = (ch.utils.html.hasClass("ie6")) ? that.itemsWidth : 0;
 64 		
 65 		// Set width to Carousel if exists a width in configuration
 66 		if (ch.utils.hasOwn(conf, "width")) { that.$element.css("width", conf.width); }
 67 		
 68 		// Set height to Carousel if exists a height in configuration
 69 		if (ch.utils.hasOwn(conf, "height")) { that.$element.css("height", conf.height); }
 70 		
 71 		// Disable CSS transition if it's specified
 72 		if (!conf.fx && ch.features.transition) { that.$content.addClass("ch-carousel-nofx"); }
 73 		
 74 		// Set container size based on items size
 75 		that.$mask.css("height", that.$items.outerHeight());
 76 		
 77 		// WAI-ARIA for items
 78 		$.each(that.$items, function (i, e) {
 79 			
 80 			// Page where this item is in
 81 			var page = ~~(i / that.itemsPerPage) + 1;
 82 			
 83 			$(e).attr({
 84 				"aria-hidden": page !== that.currentPage,
 85 				"aria-setsize": that.itemsTotal,
 86 				"aria-posinset": i + 1,
 87 				"aria-label": "page" + page
 88 			});
 89 		});
 90 		
 91 		// Total amount of items (Widthout include queue items)
 92 		var itemsAmount = that.$items.length;
 93 		
 94 		// At the begin, add items from queue if page is incomplete
 95 		if (ch.utils.hasOwn(conf, "asyncData") && itemsAmount < that.itemsPerPage) {
 96 			that.addItems(that.itemsPerPage - itemsAmount);
 97 		}
 98 	},

100 	/**
101 	* Creates Previous and Next arrows.
102 	* @private
103 	* @function
104 	* @name ch.Carousel#createArrows
105 	*/
106 		createArrows = function () {
107 			
108 			// Previous arrow
109 			var $prev = $("<p class=\"ch-prev-arrow" + (conf.rolling ? "" : " ch-hide") + "\" role=\"button\" aria-hidden=\"" + (!conf.rolling) + "\"><span>Previous</span></p>")
110 				.bind("click", that.prev)
111 				.prependTo(that.$element),
112 			
113 			// Next arrow
114 				$next = $("<p class=\"ch-next-arrow\" role=\"button\" aria-hidden=\"false\"><span>Next</span></p>")
115 				.bind("click", that.next)
116 				.appendTo(that.$element);
117 			
118 			// Positions arrows vertically in middle of Carousel
119 			$prev[0].style.top = $next[0].style.top = (that.$element.outerHeight() - $prev.outerHeight()) / 2 + "px";
120 			
121 			/**
122 			* Manages arrows turning it on and off when non-continue Carousel is moving.
123 			* @protected
124 			* @function
125 			* @name ch.Carousel#manageArrows
126 			* @param {Number} page Page to be moved.
127 			*/
128 			that.manageArrows = function (page) {
129 				// Case 1: Both arrows shown on carousel's middle
130 				if (page > 1 && page < that.pages) {
131 					$prev.attr("aria-hidden", "false").removeClass("ch-hide");
132 					$next.attr("aria-hidden", "false").removeClass("ch-hide");
133 				} else {
134 				// Case 2: Previous arrow hidden on first page
135 					if (page === 1) {
136 						$prev.addClass("ch-hide").attr("aria-hidden", "true");
137 						$next.attr("aria-hidden", "false").removeClass("ch-hide");
138 				// Case 3: Next arrow hidden on last page
139 					} else if (page === that.pages) {
140 						$prev.attr("aria-hidden", "false").removeClass("ch-hide");
141 						$next.addClass("ch-hide").attr("aria-hidden", "true");
142 					}
143 				}
144 			};
145 		},

147 	/**
148 	* Creates Carousel pagination.
149 	* @private
150 	* @function
151 	* @name ch.Carousel#createPagination
152 	*/
153 	// TODO: Re-create pagination only if amount of pages change. Else, re-position it.
154 		createPagination = function () {
155 			
156 			// Create a element List for new pagination
157 			var $pagination = $("<ul class=\"ch-carousel-pages ch-hide\" role=\"tablist\">"),
158 			
159 			// Each page into list element
160 				$thumbnails;
161 	
162 			// Create each mini thumbnail an append to list
163 			for (var i = 1; i <= that.pages; i += 1) {
164 				
165 				// Mark as active if thumbnail is the same that current page
166 				var status = (i === that.currentPage) ? " class=\"ch-carousel-pages-on\" aria-selected=\"true\"" : " aria-selected=\"false\"",
167 				
168 				// Thumbnail with closure
169 					$thumb = $("<li" + status + " role=\"tab\" aria-controls=\"page" + i + "\">" + i + "</li>")
170 						.bind("click", function (i) {
171 							return function () {
172 								that.goTo(i);
173 							};
174 						}(i));
175 				
176 				$pagination.append($thumb);
177 			};
178 			
179 			// Append list to carousel
180 			that.$element.append($pagination);
181 			
182 			// Positions list
183 			$pagination.css("left", (that.$element.outerWidth() - $pagination.outerWidth()) / 2).removeClass("ch-hide");
184 			
185 			// Set pagination children as thumbnails
186 			$thumbnails = $pagination.children();
187 			
188 			/**
189 			* Removes the active status classname of last selected page and adds it to the selected page.
190 			* @protected
191 			* @function
192 			* @name ch.Carousel#managePagination
193 			* @param {Number} page Page to be moved.
194 			*/
195 			that.managePagination = function (page) {
196 				$thumbnails.eq(that.currentPage - 1).removeClass("ch-carousel-pages-on").attr("aria-selected", "false");
197 				$thumbnails.eq(page - 1).addClass("ch-carousel-pages-on").attr("aria-selected", "true");
198 			};

201 	/**
202 	* Calculates items amount on each page.
203 	* @protected
204 	* @name ch.Carousel#getItemsPerPage
205 	* @function
206 	*/
207 		getItemsPerPage = function () {
208 			// Space to be distributed among all items
209 			var widthDiff = that.$mask.outerWidth() - that.itemsWidth;
210 			
211 			// If there are space to be distributed, calculate pages
212 			return that.itemsPerPage = (widthDiff > that.itemsWidth) ? ~~(widthDiff / that.itemsWidth) : 1;
213 		},
214 	
215 	/**
216 	* Calculates total amount of pages.
217 	* @private
218 	* @function
219 	* @name ch.Carousel#getPages
220 	* @returns {Number} Total amount of pages
221 	*/
222 		getPages = function () {
223 			// (Total amount of items) / (items amount on each page)
224 			// TODO: $coll.children =? that.$items.length
225 			return that.pages = Math.ceil(that.itemsTotal / that.itemsPerPage);
226 		},

228 	/**
229 	* Calculates all necesary data to draw Carousel correctly.
230 	* @private
231 	* @function
232 	* @name ch.Carousel#draw
233 	*/
234 		draw = function () {
235 			
236 			// Reset size of carousel mask
237 			maskWidth = that.$mask.outerWidth();
238 			
239 			// Recalculate items amount on each page
240 			getItemsPerPage();
241 			
242 			// Recalculate total amount of pages
243 			getPages();
244 			
245 			// Calculate variable margin between each item
246 			that.itemsMargin = Math.ceil(((maskWidth - (that.itemsWidth * that.itemsPerPage)) / that.itemsPerPage) / 2);
247 			
248 			// Modify sizes only if new items margin are positive numbers
249 			if (that.itemsMargin < 0) { return; }
250 			
251 			// Detach content from DOM for make a few changes
252 			that.$content.detach();
253 			
254 			// Move Carousel to first page for reset initial position
255 			that.goTo(1);
256 			
257 			// Save rendered items amount
258 			var i = that.$items.length;
259 			
260 			// Set new margin to all items
261 			while (i) {
262 				that.$items[i -= 1].style.marginLeft = that.$items[i].style.marginRight = that.itemsMargin + "px";
263 			}
264 			
265 			// Change content size and append it to DOM again
266 			// TODO: Use "width:-moz-max-content;" once instead .css("width"). Maybe add support to ch.features
267 			that.$content
268 				.css("width", (that.itemsWidth + that.itemsMargin * 2) * that.$items.length + extraWidth)
269 				.appendTo(that.$mask);
270 			
271 			// Manage Previous and Next arrows
272 			if (conf.arrows) {
273 				// Deletes pagination if already exists
274 				that.$element.find(".ch-prev-arrow, .ch-next-arrow").remove();
275 				
276 				// Creates updated pagination
277 				if (that.pages > 1) { createArrows(); }
278 			}
279 			
280 			// Manage pagination
281 			if (conf.pagination) {
282 				// Deletes pagination if already exists
283 				that.$element.find(".ch-carousel-pages").remove();
284 				
285 				// Creates updated pagination
286 				if (that.pages > 1) { createPagination(); }
287 			}
288 		},

408 	/**
409 	* Move items from queue to collection.
410 	* @protected
411 	* @name ch.Carousel#addItems
412 	* @function
413 	* @param {Number} amount Amount of items that will be added.
414 	*/
415 	that.addItems = function (amount) {
416 		
417 		// Take the sample from queue
418 		var sample = that.queue.splice(0, amount),
419 		
420 		// Condition if exists a render function on component configuration object
421 			hasRender = ch.utils.hasOwn(conf, "asyncRender"),
422 		
423 		// Position where new items will be added
424 			itemIndex = that.$items.length;
425 		
426 		// Append asynchronous items to collection
427 		// HTML Li Element with classname and styles and content from conf.async with or without render function
428 		for (var i = 0; i < amount; i += 1) {
429 			
430 			// Page where this item is in
431 			var page = ~~(itemIndex / that.itemsPerPage) + 1;
432 			
433 			sample[i] = "<li class=\"ch-carousel-item\" role=\"listitem\" aria-hidden=\"" + (page !== that.currentPage) + "\" aria-setsize=\"" + that.itemsTotal + "\" aria-posinset=\"" + (itemIndex += 1) + "\" aria-label=\"page" + page + "\" style=\"margin-right: " + that.itemsMargin + "px; margin-left: " + that.itemsMargin + "px;\">" + (hasRender ? conf.asyncRender(sample[i]) : sample[i]) + "</li>";
434 		};
435 		
436 		// Expand content width for include new items (item width and margin) * (total amount of items) + extra width
437 		// TODO: Use "width:-moz-max-content;" once instead .css("width"). Maybe add support to ch.features
438 		that.$content.css("width", (that.itemsWidth + that.itemsMargin * 2) * that.itemsTotal + extraWidth);
439 		
440 		// Append collection again
441 		that.$collection.append(sample.join(""));
442 		
443 		// Update items collection
444 		that.$items = that.$collection.children();
445 		
446 		/**
447 		* Callback function
448 		* @name ch.Carousel#onItemsAdded
449 		* @type Function
450 		*/
451 		that.callbacks("onItemsAdded");
452 		// new callback
453 		that.trigger("itemsAdded");
454 		
455 	};
456 	
457 	
458 	/**
459 	* Analizes if next page needs to load items from queue and execute addItems() method.
460 	* @protected
461 	* @name ch.Carousel#asyncItemsLoad
462 	* @function
463 	*/
464 	that.asyncItemsLoad = function () {
465 		
466 		// Load only when there are items in queue
467 		if (that.queue.length === 0) { return; }
468 		
469 		// Amount of items from the beginning to current page
470 		var itemsHere = that.currentPage * that.itemsPerPage,
471 		
472 		// Items rendered
473 			itemsRendered = that.$items.length;
474 		
475 		// Load only when there are more visible items than items rendered
476 		if (itemsHere < itemsRendered) { return; }
477 		
478 		// How many items needs to add for complete next page
479 		var amount = itemsHere % itemsRendered;
480 		
481 		// If isn't needed items to complete a page, then add an entire page
482 		amount = (amount === 0) ? that.itemsPerPage : amount;
483 		
484 		// If next page needs less items than it support, then add that amount
485 		amount = (that.queue.length < amount) ? that.queue.length : amount;
486 		
487 		// Add these
488 		that.addItems(amount);
489 		
490 	};
491 	
492 	// Moves to a defined page
493 	that.goTo = function (page) {
494 		
495 		// Validation of page parameter
496 		if (page === that.currentPage || page > that.pages || page < 1 || isNaN(page)) { return that; }
497 		
498 		// Manage arrows
499 		if (!conf.rolling && conf.arrows) { that.manageArrows(page); }
500 		
501 		// Select thumbnail on pagination
502 		if (conf.pagination) { that.managePagination(page); }
503 		
504 		// Coordinates of next movement
505 		var movement = -(maskWidth * (page - 1));
506 
507 		// TODO: review this conditional
508 		// Case 1: Movement with CSS transition
509 		if (conf.fx && ch.features.transition) {
510 			that.$content.css("left", movement);
511 		// Case 2: Movement with jQuery animate
512 		} else if (conf.fx) {
513 			that.$content.animate({ left: movement });
514 		// Case 3: Movement without transition or jQuery
515 		} else {
516 			that.$content.css("left", movement);
517 		}
518 		
519 		// Refresh selected page
520 		that.currentPage = page;
521 		
522 		// WAI-ARIA to set items as "hide"
523 		$.each(that.$items, function (i, e) {
524 			$(e).attr("aria-hidden", ~~(i / that.itemsPerPage) + 1 !== page);
525 		});
526 		
527 		/**
528 		* Callback function
529 		* @name ch.Carousel#onMove
530 		* @type Function
531 		*/
532 		that.callbacks("onMove");
533 		// new callback
534 		that.trigger("move");
535 		
536 		return that;
537 	};

666 	/**
667 	* Gets the current page or moves to a defined page.
668 	* @public
669 	* @function
670 	* @name ch.Carousel#page
671 	* @returns Chico-UI Object
672 	* @param {Number} page Page to be moved.
673 	* @since 0.7.4
674 	* @example
675 	* // Create a carousel
676 	* var foo = $("bar").carousel();
677 	* 
678 	* // Go to second page
679 	* foo.page(2);
680 	* @example
681 	* // Get the current page
682 	* foo.page();
683 	*/
684 	that["public"].page = function (data) {
685 		// Getter
686 		if (!data) { return that.currentPage; }
