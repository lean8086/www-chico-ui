1 /**
  2 * Zoom shows a contextual reference to an augmented version of main declared image.
  3 * @name Zoom
  4 * @class Zoom
  5 * @augments ch.Floats
  6 * @requires ch.Positioner
  7 * @requires ch.onImagesLoads
  8 * @memberOf ch
  9 * @param {Object} [conf] Object with configuration properties.
 10 * @param {Boolean} [conf.fx] Enable or disable fade effect on show. By default, the effect are disabled.
 11 * @param {Boolean} [conf.context] Sets a reference to position of component that will be considered to carry out the position. By default is the anchor of HTML snippet.
 12 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or "lt rt" by default.
 13 * @param {String} [conf.offset] Sets the offset in pixels that component will be displaced from original position determined by points. It's specified by configuration or "20 0" by default.
 14 * @param {String} [conf.message] This message will be shown when component needs to communicate that is in process of load. It's "Loading zoom..." by default.
 15 * @param {Number} [conf.width] Width of floated area of zoomed image. Example: 500, "500px", "50%". Default: 350.
 16 * @param {Number} [conf.height] Height of floated area of zoomed image. Example: 500, "500px", "50%". Default: 350.
 17 * @returns itself
 18 * @exampleDescription Create a Zoom component wrapping the original image with a anchor element pointing to a bigger version than the original.
 19 * @example
 20 * var widget = $(".example").zoom();
 21 * @factorized
 22 * @see ch.Modal
 23 * @see ch.Tooltip
 24 * @see ch.Layer
 25 */
 26 
 27 ch.zoom = function (conf) {
 28 	/**
 29 	* Reference to an internal component instance, saves all the information and configuration properties.
 30 	* @private
 31 	* @name ch.Zoom#that
 32 	* @type itself
 33 	*/
 34 	var that = this;
 35 
 36 /**
 37 *	Constructor
 38 */
 39 	conf = ch.clon(conf);
 40 
 41 	conf.fx = conf.fx || false;
 42 	
 43 	conf.cache = false;
 44 
 45 	// WAI-ARIA
 46 	conf.aria = {};
 47 	conf.aria.role = "tooltip";
 48 	conf.aria.identifier = "aria-describedby";
 49 
 50 	// Position
 51 	conf.position = {};
 52 	conf.position.context = conf.context || that.$element;
 53 	conf.position.offset = conf.offset || "20 0";
 54 	conf.position.points = conf.points || "lt rt";
 55 	conf.reposition = false;
 56 
 57 	// Transition message and size
 58 	conf.message = conf.message || "Loading zoom...";
 59 	conf.width = conf.width || 300;
 60 	conf.height = conf.height || 300;
 61 
 62 	// Closable configuration
 63 	conf.closable = false;
 64 
 65 	that.conf = conf;
 66 
 67 /**
 68 *	Inheritance
 69 */
 70 
 71 	that = ch.floats.call(that);
 72 	that.parent = ch.clon(that);

 74 /**
 75 *	Private Members
 76 */
 77 
 78 	/**
 79 	* Element showed before zoomed image is load. It's a transition message and its content can be configured through parameter "message".
 80 	* @private
 81 	* @name ch.Zoom#$loading
 82 	* @type Object
 83 	*/
 84 	var $loading = $("<p class=\"ch-zoom-loading ch-hide\">" + conf.message + "</p>").appendTo(that.$element),
 85 
 86 	/**
 87 	* Position of main anchor. It's for calculate cursor position hover the image.
 88 	* @private
 89 	* @name ch.Zoom#offset
 90 	* @type Object
 91 	*/
 92 		offset = that.$element.offset(),
 93 
 94 	/**
 95 	* Visual element that follows mouse movement for reference to zoomed area into original image.
 96 	* @private
 97 	* @name ch.Zoom#seeker
 98 	* @type Object
 99 	*/
100 		seeker = {
101 			/**
102 			* Element shown as seeker.
103 			* @private
104 			* @name shape
105 			* @memberOf ch.Zoom#seeker
106 			* @type Object
107 			*/
108 			"$shape": $("<div class=\"ch-zoom-seeker ch-hide\">"),
109 
110 			/**
111 			* Half of width of seeker element. It's only half to facilitate move calculations.
112 			* @private
113 			* @name width
114 			* @memberOf ch.Zoom#seeker
115 			* @type Number
116 			*/
117 			"width": 0,
118 
119 			/**
120 			* Half of height of seeker element. It's only half to facilitate move calculations.
121 			* @private
122 			* @name height
123 			* @memberOf ch.Zoom#seeker
124 			* @type Number
125 			*/
126 			"height": 0
127 		},
128 
129 	/**
130 	* Reference to main image declared on HTML code snippet.
131 	* @private
132 	* @name ch.Zoom#original
133 	* @type Object
134 	*/
135 		original = (function () {
136 			// Define the content source
137 			var $img = that.$element.children("img");
138 
139 			// Grab some data when image loads
140 			$img.onImagesLoads(function () {
141 
142 				// Grab size of original image
143 				original.width = $img.prop("width");
144 				original.height = $img.prop("height");
145 
146 				// Anchor size (same as image)
147 				that.$element.css({
148 					"width": original.width,
149 					"height": original.height
150 				});
151 
152 				// Loading position centered at anchor
153 				$loading.css({
154 					"left": (original.width - $loading.width()) / 2,
155 					"top": (original.height - $loading.height()) / 2
156 				});
157 
158 			});
159 
160 			return {
161 				/**
162 				* Reference to HTML Element of original image.
163 				* @private
164 				* @name img
165 				* @memberOf ch.Zoom#original
166 				* @type Object
167 				*/
168 				"$image": $img,
169 
170 				/**
171 				* Position of original image relative to viewport.
172 				* @private
173 				* @name offset
174 				* @memberOf ch.Zoom#original
175 				* @type Object
176 				*/
177 				"offset": {},
178 
179 				/**
180 				* Width of original image.
181 				* @private
182 				* @name width
183 				* @memberOf ch.Zoom#original
184 				* @type Number
185 				*/
186 				"width": 0,
187 
188 				/**
189 				* Height of original image.
190 				* @private
191 				* @name height
192 				* @memberOf ch.Zoom#original
193 				* @type Number
194 				*/
195 				"height": 0
196 			};
197 		}()),
198 
199 	/**
200 	* Relative size between zoomed and original image.
201 	* @private
202 	* @name ch.Zoom#ratio
203 	* @type Object
204 	*/
205 		ratio = {
206 			/**
207 			* Relative size of X axis.
208 			* @private
209 			* @name width
210 			* @memberOf ch.Zoom#ratio
211 			* @type Number
212 			*/
213 			"width": 0,
214 
215 			/**
216 			* Relative size of Y axis.
217 			* @private
218 			* @name height
219 			* @memberOf ch.Zoom#ratio
220 			* @type Number
221 			*/
222 			"height": 0
223 		},
224 
225 	/**
226 	* Reference to the augmented version of image, that will be displayed into a floated element.
227 	* @private
228 	* @name ch.Zoom#zoomed
229 	* @type Object
230 	*/
231 		zoomed = (function () {
232 			// Define the content source
233 			var $img = that.source = $("<img src=\"" + that.element.href + "\">");
234 
235 			// Grab some data when zoomed image loads
236 			$img.onImagesLoads(function () {
237 
238 				// Save the zoomed image size
239 				zoomed.width = $img.prop("width");
240 				zoomed.height = $img.prop("height");
241 
242 				// Save the zoom ratio
243 				ratio.width = zoomed.width / original.width;
244 				ratio.height = zoomed.height / original.height;
245 
246 				// Seeker: Size relative to zoomed image respect zoomed area
247 				var w = ~~(conf.width / ratio.width),
248 					h = ~~(conf.height / ratio.height);
249 
250 				// Seeker: Save half width and half height
251 				seeker.width = w / 2;
252 				seeker.height = h / 2;
253 
254 				// Seeker: Set size and append it
255 				seeker.$shape.css({"width": w, "height": h}).appendTo(that.$element);
256 
257 				// Remove loading
258 				$loading.remove();
259 
260 				// Change zoomed image status to Ready
261 				zoomed.ready = true;
262 
263 				// TODO: MAGIC here! if mouse is over image show seeker and make all that innerShow do
264 			});
265 
266 			return {
267 				/**
268 				* Reference to HTML Element of augmented image.
269 				* @private
270 				* @name img
271 				* @memberOf ch.Zoom#zoomed
272 				* @type Object
273 				*/
274 				"$image": $img,
275 
276 				/**
277 				* Status of augmented image. When it's load, the status is "true".
278 				* @private
279 				* @name ready
280 				* @memberOf ch.Zoom#zoomed
281 				* @type Boolean
282 				*/
283 				"ready": false,
284 
285 				/**
286 				* Width of augmented image.
287 				* @private
288 				* @name width
289 				* @memberOf ch.Zoom#zoomed
290 				* @type Number
291 				*/
292 				"width": 0,
293 
294 				/**
295 				* Height of augmented image.
296 				* @private
297 				* @name height
298 				* @memberOf ch.Zoom#zoomed
299 				* @type Number
300 				*/
301 				"height": 0
302 			};
303 		}()),
304 
305 	/**
306 	* Calculates movement limits and sets it to seeker and augmented image.
307 	* @private
308 	* @function
309 	* @name ch.Zoom#move
310 	* @param {Event} event Mouse event to take the cursor position.
311 	*/
312 		move = function (event) {
313 
314 			var x, y;
315 
316 			// Left side of seeker LESS THAN left side of image
317 			if (event.pageX - seeker.width < offset.left) {
318 				x = 0;
319 			// Right side of seeker GREATER THAN right side of image
320 			} else if (event.pageX + seeker.width > original.width + offset.left) {
321 				x = original.width - (seeker.width * 2) - 2;
322 			// Free move
323 			} else {
324 				x = event.pageX - offset.left - seeker.width;
325 			}
326 
327 			// Top side of seeker LESS THAN top side of image
328 			if (event.pageY - seeker.height < offset.top) {
329 				y = 0;
330 			// Bottom side of seeker GREATER THAN bottom side of image
331 			} else if (event.pageY + seeker.height > original.height + offset.top) {
332 				y = original.height - (seeker.height * 2) - 2;
333 			// Free move
334 			} else {
335 				y = event.pageY - offset.top - seeker.height;
336 			}
337 
338 			// Move seeker
339 			seeker.$shape.css({"left": x, "top": y});
340 
341 			// Move zoomed image
342 			zoomed.$image.css({"left": (-ratio.width * x), "top": (-ratio.height * y)});
343 
344 		};

350 	/**
351 	* Inner show method. Attach the component's layout to the DOM tree and load defined content.
352 	* @protected
353 	* @name ch.Zoom#innerShow
354 	* @function
355 	* @returns itself
356 	*/
357 	that.innerShow = function () {
358 
359 		// If the component isn't loaded, show loading transition
360 		if (!zoomed.ready) {
361 			$loading.removeClass("ch-hide");
362 			return that;
363 		}
364 
365 		// Update position of anchor here because Zoom can be inside a Carousel and its position updates
366 		offset = that.$element.offset();
367 
368 		// Bind move calculations
369 		that.$element.bind("mousemove", function (event) { move(event); });
370 
371 		// Show seeker
372 		seeker.$shape.removeClass("ch-hide");
373 
374 		// Show float
375 		that.parent.innerShow();
376 
377 		return that;
378 
379 	};
380 
381 	/**
382 	* Inner hide method. Hides the component's layout and detach it from DOM tree.
383 	* @protected
384 	* @name ch.Zoom#innerHide
385 	* @function
386 	* @returns itself
387 	*/
388 	that.innerHide = function () {
389 
390 		// If the component isn't loaded, hide loading transition
391 		if (!zoomed.ready) {
392 			$loading.addClass("ch-hide");
393 			return that;
394 		}
395 
396 		// Unbind move calculations
397 		that.$element.unbind("mousemove");
398 
399 		// Hide seeker
400 		seeker.$shape.addClass("ch-hide");
401 
402 		// Hide float
403 		that.parent.innerHide();
404 
405 		return that;
406 
407 	};
408 
409 	/**
410 	* Getter and setter for size attributes of float that contains the zoomed image.
411 	* @protected
412 	* @function
413 	* @name ch.Zoom#size
414 	* @param {string} prop Property that will be setted or getted, like "width" or "height".
415 	* @param {string} [data] Only for setter. It's the new value of defined property.
416 	* @returns itself
417 	*/
418 	that.size = function (prop, data) {
419 
420 		// Seeker: Updates styles and size value
421 		if (data) {
422 			// Seeker: Size relative to zoomed image respect zoomed area
423 			var size = ~~(data / ratio[prop]);
424 
425 			// Seeker: Save half width and half height
426 			seeker[prop] = size / 2;
427 
428 			// Seeker: Set size
429 			seeker.$shape.css(prop, size);
430 		}
431 
432 		// Change float size
433 		return that.parent.size(prop, data);
434 	};
