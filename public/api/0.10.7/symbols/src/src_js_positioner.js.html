1 /**
  2 * Positioner lets you centralize and manage changes related to positioned elements. Positioner returns an utility that resolves positioning for all widget.
  3 * @name Positioner
  4 * @class Positioner
  5 * @memberOf ch
  6 * @param {Object} conf Configuration object with positioning properties.
  7 * @param {String} conf.element Reference to the DOM Element to be positioned.
  8 * @param {String} [conf.context] It's a reference to position and size of element that will be considered to carry out the position. If it isn't defined through configuration, it will be the viewport.
  9 * @param {String} [conf.points] Points where element will be positioned, specified by configuration or centered by default.
 10 * @param {String} [conf.offset] Offset in pixels that element will be displaced from original position determined by points. It's specified by configuration or zero by default.
 11 * @param {Boolean} [conf.reposition] Parameter that enables or disables reposition intelligence. It's disabled by default.
 12 * @requires ch.Viewport
 13 * @see ch.Viewport
 14 * @returns {Function} The Positioner returns a Function that it works in 3 ways: as a setter, as a getter and with the "refresh" parameter refreshes the position.
 15 * @exampleDescription
 16 * Instance the Positioner It requires a little configuration. 
 17 * The default behavior place an element centered into the Viewport. 
 18 *  
 19 * @example
 20 * var positioned = ch.positioner({
 21 *     element: "#element1",
 22 * });
 23 * @exampleDescription 1. Getting the current configuration properties.
 24 * @example
 25 * var configuration = positioned()
 26 * @exampleDescription 2. Updates the current position with <code>refresh</code> as a parameter. 
 27 * @example
 28 * positioned("refresh");
 29 * @exampleDescription 3. Define a new position
 30 * @example
 31 * positioned({
 32 *     element: "#element2",
 33 *     context: "#context2",
 34 *     points: "lt rt"
 35 * });
 36 * @exampleDescription <strong>Offset</strong>: The positioner could be configurated with an offset. 
 37 * This example show an element displaced horizontally by 10px of defined position.
 38 * @example
 39 * var positioned = ch.positioner({
 40 *     element: "#element3",
 41 *     context: "#context3",
 42 *     points: "lt rt",
 43 *     offset: "10 0"
 44 * });
 45 * @exampleDescription <strong>Reposition</strong>: Repositionable feature moves the postioned element if it can be shown into the viewport.
 46 * @example
 47 * var positioned = ch.positioner({
 48 *     element: "#element4",
 49 *     context: "#context4",
 50 *     points: "lt rt",
 51 *     reposition: true
 52 * });

425 		/**
426 		* Calculates left and top position from specific points.
427 		* @private
428 		* @name ch.Positioner#getCoordinates
429 		* @function
430 		* @param {String} points String with points to be calculated.
431 		* @returns Offset measures
432 		* @exampleDescription
433 		* @example
434 		* var foo = getCoordinates("lt rt");
435 		* 
436 		* foo = {
437 		*     left: Number,
438 		*     top: Number
439 		* };
440 		*/
441 			getCoordinates = function (pts) {
442 
443 				// Calculates left or top position from points related to specific axis (X or Y).
444 				// TODO: Complete cases: X -> lc, cl, rc, cr. Y -> tm, mt, bm, mb.
445 				var calculate = function (reference) {
446 
447 					// Use Position or Offset of Viewport if position is fixed or absolute respectively
448 					var ctx = (!contextIsNotViewport && ch.features.fixed) ? ch.viewport.getPosition() : context,
449 					
450 					// Returnable value
451 						r;
452 
453 					switch (reference) {
454 					// X references
455 					case "ll": r = ctx.left + offset[0]; break;
456 					case "lr": r = ctx.right + offset[0]; break;
457 					case "rl": r = ctx.left - $element.outerWidth() + offset[0]; break;
458 					case "rr": r = ctx.right - $element.outerWidth() + offset[0]; break;
459 					case "cc": r = ctx.left + (ctx.width / 2) - ($element.outerWidth() / 2) + offset[0]; break;
460 					// Y references
461 					case "tt": r = ctx.top + offset[1]; break;
462 					case "tb": r = ctx.bottom + offset[1]; break;
463 					case "bt": r = ctx.top - $element.outerHeight() + offset[1]; break;
464 					case "bb": r = ctx.bottom - $element.outerHeight() + offset[1]; break;
465 					case "mm": r = ctx.top + (ctx.height / 2) - ($element.outerHeight() / 2) + offset[1]; break;
466 					}
467 
468 					return r;
469 				},
470 
471 				// Splitted points
472 					splittedPoints = pts.split(" ");
473 
474 				// Calculates left and top with references to X and Y axis points (crossed points)
475 				return {
476 					"left": calculate(splittedPoints[0].charAt(0) + splittedPoints[1].charAt(0)),
477 					"top": calculate(splittedPoints[0].charAt(1) + splittedPoints[1].charAt(1))
478 				};
479 			},

542 						classPoints = newData.classPoints;
543 					}
