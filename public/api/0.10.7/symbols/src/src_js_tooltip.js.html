1 /**
  2 * Tooltip improves the native tooltips. Tooltip uses the 'alt' and 'title' attributes to grab its content.
  3 * @name Tooltip
  4 * @class Tooltip
  5 * @augments ch.Floats
  6 * @memberOf ch
  7 * @param {Object} [conf] Object with configuration properties.
  8 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are enable.
  9 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or centered by default: "cm cm".
 10 * @param {String} [conf.offset] Sets the offset in pixels that component will be displaced from original position determined by points. It's specified by configuration or zero by default: "0 0".
 11 * @returns itself
 12 * @factorized
 13 * @see ch.Modal
 14 * @see ch.Layer
 15 * @see ch.Zoom
 16 * @exampleDescription Create a tooltip.
 17 * @example
 18 * var widget = $(".some-element").tooltip();
 19 * @exampleDescription Create a new tooltip with configuration.
 20 * @example
 21 * var widget = $("a.example").tooltip({
 22 *     "fx": false,
 23 *     "offset": "10 -10",
 24 *     "points": "lt rt"
 25 * });
 26 * @exampleDescription
 27 * Now <code>widget</code> is a reference to the tooltip instance controller.
 28 * You can set a new content by using <code>widget</code> like this: 
 29 * @example
 30 * widget.width(300);
 31 */
 32 
 33 ch.tooltip = function (conf) {
 34 
 35 	/**
 36 	* Reference to a internal component instance, saves all the information and configuration properties.
 37 	* @private
 38 	* @name ch.Tooltip#that
 39 	* @type object
 40 	*/
 41 	var that = this;
 42 
 43 	conf = ch.clon(conf);
 44 
 45 	conf.cone = true;
 46 	conf.content = that.element.title || that.element.alt;
 47 	
 48 	// Closable configuration
 49 	conf.closable = false;
 50 
 51 	conf.aria = {};
 52 	conf.aria.role = "tooltip";
 53 	conf.aria.identifier = "aria-describedby";
 54 
 55 	conf.position = {};
 56 	conf.position.context = that.$element;
 57 	conf.position.offset = conf.offset || "0 10";
 58 	conf.position.points = conf.points || "lt lb";
 59 
 60 	that.conf = conf;
 61 
 62 /**
 63 *	Inheritance
 64 */
 65 
 66 	that = ch.floats.call(that);
 67 	that.parent = ch.clon(that);
 68 
 69 /**
 70 *	Private Members
 71 */
 72 	/**
 73 	* The attribute that will provide the content. It can be "title" or "alt" attributes.
 74 	* @protected
 75 	* @name ch.Tooltip#attrReference
 76 	* @type string
 77 	*/
 78 	var attrReference = (that.element.title) ? "title" : "alt",
 79 
 80 	/**
 81 	* The original attribute content.
 82 	* @private
 83 	* @name ch.Tooltip#attrContent
 84 	* @type string
 85 	*/
 86 		attrContent = that.element.title || that.element.alt;
 87 
 88 /**
 89 *	Protected Members
 90 */
 91 
 92 	/**
 93 	* Inner show method. Attach the component layout to the DOM tree.
 94 	* @protected
 95 	* @name ch.Tooltip#innerShow
 96 	* @function
 97 	* @returns itself
 98 	*/
 99 	that.innerShow = function (event) {
100 		
101 		// Reset all tooltip, except me
102 		$.each(ch.instances.tooltip, function (i, e) {
103 			if (e !== that["public"]) {
104 				e.hide();
105 			}
106 		});
107 		
108 		// IE8 remembers the attribute even when is removed, so ... empty the attribute to fix the bug.
109 		that.element[attrReference] = "";
110 
111 		that.parent.innerShow(event);
112 
113 		return that;
114 	};
115 
116 	/**
117 	* Inner hide method. Hides the component and detach it from DOM tree.
118 	* @protected
119 	* @name ch.Tooltip#innerHide
120 	* @function
121 	* @returns itself
122 	*/
123 	that.innerHide = function (event) {
124 		that.element[attrReference] = attrContent;
125 
126 		that.parent.innerHide(event);
127 
128 		return that;
129 	};
130 
131 /**
132 *	Public Members
133 */
134 
135 	/**
136 	* @borrows ch.Object#uid as ch.Tooltip#uid
137 	*/	
138 	
139 	/**
140 	* @borrows ch.Object#element as ch.Tooltip#element
141 	*/
142 
143 	/**
144 	* @borrows ch.Object#type as ch.Tooltip#type
145 	*/
146 
147 	/**
148 	* @borrows ch.Object#content as ch.Tooltip#content
149 	*/
150 
151 	/**
152 	* @borrows ch.Floats#isActive as ch.Tooltip#isActive
153 	*/
154 
155 	/**
156 	* @borrows ch.Floats#show as ch.Tooltip#show
157 	*/
158 
159 	/**
160 	* @borrows ch.Floats#hide as ch.Tooltip#hide
161 	*/
162 
163 	/**
164 	* @borrows ch.Floats#width as ch.Tooltip#width
165 	*/
166 
167 	/**
168 	* @borrows ch.Floats#height as ch.Tooltip#height
169 	*/
170 
171 	/**
172 	* @borrows ch.Floats#position as ch.Tooltip#position
173 	*/
174 
175 	/**
176 	* @borrows ch.Floats#closable as ch.Tooltip#closable
177 	*/
178 
179 /**
180 *	Default event delegation
181 */
182 
183 	that.$element
184 		.bind("mouseenter", that.innerShow)
185 		.bind("mouseleave", that.innerHide);
186 	
187 	/**
188 	* Triggers when component is ready to use.
189 	* @name ch.Tooltip#ready
190 	* @event
191 	* @public
192 	* @example
193 	* // Following the first example, using <code>widget</code> as tooltip's instance controller:
194 	* widget.on("ready",function () {
195 	*	this.show();
196 	* });
197 	*/
198 	setTimeout(function(){ that.trigger("ready")}, 50);
199 
200 	return that;
201 };
202 
203 ch.factory("tooltip");
