1 (function ($, ch) {
  2 	'use strict';
  3 
  4 	// Initial options to be merged with the user's options
  5 	var defaults = {
  6 		'method': 'GET',
  7 		'params': '',
  8 		'cache': true,
  9 		'async': true
 10 	};
 11 
 12 	/**
 13 	 * Creates a component to manage content through 3 ways: plain text, DOM elements, AJAX requests.
 14 	 * @memberOf ch
 15 	 * @class ch.Content
 16 	 * @require ch.cache
 17 	 * @returns {Object}
 18 	 */
 19 	function Content() {
 20 
 21 		var that = this,
 22 			// Merged options of each instance
 23 			options,
 24 			// The lastest data sent to the client. Used to return on the .get() method
 25 			current = 'Chico Error: Content is not defined.',
 26 			/**
 27 			 * Allows to manage the widgets content.
 28 			 * @namespace
 29 			 */
 30 			content = function ($content) {
 31 				// Gets a new content
 32 				if ($content === undefined) {
 33 					return that.content.get();
 34 				}
 35 
 36 				// Configures a new content
 37 				that.content.configure({
 38 					'input': $content
 39 				});
 40 
 41 				// Sets the new content only if the component is active
 42 				if (that.active) {
 43 					that.content.set();
 44 				}
 45 
 46 				return that["public"];
 47 
 48 			};
 49 
 50 		/**
 51 		 * Send the result data to the "client".
 52 		 * @private
 53 		 */
 54 		function postMessage(data) {
 55 			// Update the lastest reference of data sent to the user
 56 			content.onmessage(current = data);
 57 			// TODO: Trigger the "message" event to allow multiple suscription
 58 			// that.trigger('message', data)
 59 		}
 60 
 61 		/**
 62 		 * Serves data from cache or AJAX.
 63 		 * @private
 64 		 */
 65 		function getContentFromAJAX() {
 66 
 67 			// When exists posibilities of find something saved into ch.cache...
 68 			if (options.cache) {
 69 				// Try to get data from cache
 70 				var cached = ch.cache.get(options.input);
 71 				// If there are data, then send to the client and avoid the AJAX request
 72 				if (cached) { return postMessage(cached); }
 73 			}
 74 
 75 			$.ajax({
 76 				'url': options.input,
 77 				'type': options.method,
 78 				'data': 'x=x' + ((options.params !== '') ? '&' + options.params : ''),
 79 				'cache': options.cache,
 80 				'async': options.async,
 81 				'beforeSend': function (jqXHR) {
 82 					// Set the AJAX default HTTP headers
 83 					jqXHR.setRequestHeader('X-Requested-With', 'XMLHttpRequest');
 84 				},
 85 				'success': function (data) {
 86 					// Grab the data on the cache if it's necessary
 87 					if (options.cache) { ch.cache.set(options.input, data); }
 88 					// Send the result data to the client
 89 					postMessage(data);
 90 				},
 91 				'error': function (jqXHR, textStatus, errorThrown) {
 92 					// Grab all the parameters into a JSON to send to the client
 93 					var data = {
 94 						'jqXHR': jqXHR,
 95 						'textStatus': textStatus,
 96 						'errorThrown': errorThrown
 97 					};
 98 					// Send a defined error message
 99 					postMessage('<p>Error on ajax call.</p>');
100 					// Execute the 'onerror' method if exists
101 					if (ch.util.hasOwn(content, 'onerror')) {
102 						content.onerror(data);
103 					}
104 					// TODO: Trigger the "error" event to allow multiple suscription
105 					//that.trigger('error', data);
106 				}
107 			});
108 		}
109 
110 		/**
111 		 * Merges the current options with the options specified by user.
112 		 * @name configure
113 		 * @methodOf content
114 		 * @param {Object} userOptions Options specified by user.
115 		 */
116 		content.configure = function (userOptions) {
117 			// Merge the defaults options with user options
118 			options = $.extend(userOptions, defaults);
119 			// Since second time, just merge the current options with user options
120 			content.configure = function (userOptions) {
121 
122 				// Getter: return the current configuration (options)
123 				if (userOptions === undefined) {
124 					return options;
125 				}
126 
127 				// Setter: Merge current options with the new ones
128 				$.extend(options, userOptions);
129 
130 				return content;
131 			};
132 
133 			return content;
134 		};
135 
136 		/**
137 		 * Determines what kind of input have to use, and send to client plain text, DOM element or the result of an AJAX request.
138 		 * @name set
139 		 * @methodOf content
140 		 */
141 		content.set = function (userOptions) {
142 
143 			if (userOptions !== undefined) {
144 				content.configure(userOptions);
145 			}
146 
147 			// Input as string
148 			if (typeof options.input === 'string') {
149 				// Case 1: AJAX call
150 				if (ch.util.isUrl(options.input)) {
151 					getContentFromAJAX();
152 				// Case 2: Plain text
153 				} else {
154 					postMessage(options.input);
155 				}
156 			// Case 3: DOM element
157 			} else if (options.input instanceof $) {
158 				postMessage(options.input.detach().removeClass('ch-hide'));
159 			// Default: No message
160 			} else {
161 				postMessage(current);
162 			}
163 
164 			return content;
165 		};
166 
167 		/**
168 		 * Returns the last updated content.
169 		 * @name get
170 		 * @methodOf content
171 		 */
172 		content.get = function () {
173 			return current;
174 		};
175 
176 		that.content = that['public'].content = content;
177 
178 	}
179 
180 	ch.Content = Content;
181 
182 }(this.jQuery, this.ch));
