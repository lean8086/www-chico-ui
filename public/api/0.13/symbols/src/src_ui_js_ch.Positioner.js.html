  3 * @name Positioner

 10 * @param {String} [conf.offset] Offset in pixels that element will be displaced from original position determined by points. It's specified by configuration or zero by default.
 11 * @param {Boolean} [conf.reposition] Parameter that enables or disables reposition intelligence. It's disabled by default.

 13 * @see ch.Viewport
 14 * @returns {Function} The Positioner returns a Function that it works in 3 ways: as a setter, as a getter and with the "refresh" parameter refreshes the position.
 15 * @exampleDescription
 16 * Instance the Positioner It requires a little configuration.

 18 *
 19 * @example
 20 * var positioned = ch.positioner({
 21 *     element: "#element1",
 22 * });
 23 * @exampleDescription 1. Getting the current configuration properties.
 24 * @example
 25 * var configuration = positioned()
 26 * @exampleDescription 2. Updates the current position with <code>refresh</code> as a parameter.
 27 * @example
 28 * positioned("refresh");
 29 * @exampleDescription 3. Define a new position
 30 * @example
 31 * positioned({
 32 *     element: "#element2",
 33 *     context: "#context2",
 34 *     points: "lt rt"
 35 * });

432 		/**
433 		 * Calculates left and top position from specific points.
434 		 * @private
435 		 * @name ch.Positioner#getCoordinates
436 		 * @function
437 		 * @param {String} points String with points to be calculated.
438 		 * @returns Offset measures
439 		 * @exampleDescription
440 		 * @example
441 		 * var foo = getCoordinates("lt rt");

446 		 * };

448 			getCoordinates = function (pts) {
449 
450 				// Calculates left or top position from points related to specific axis (X or Y).
451 				// TODO: Complete cases: X -> lc, cl, rc, cr. Y -> tm, mt, bm, mb.
452 				var calculate = function (reference) {
453 
454 					// Use Position or Offset of Viewport if position is fixed or absolute respectively
455 					var ctx = (!contextIsNotViewport && ch.support.fixed) ? ch.viewport.getPosition() : context,
456 
457 					// Returnable value
458 						r;
459 
460 					switch (reference) {
461 					// X references
462 					case "ll": r = ctx.left + offset[0]; break;
463 					case "lr": r = ctx.right + offset[0]; break;
464 					case "rl": r = ctx.left - $element.outerWidth() + offset[0]; break;
465 					case "rr": r = ctx.right - $element.outerWidth() + offset[0]; break;
466 					case "cc": r = ctx.left + (ctx.width / 2) - ($element.outerWidth() / 2) + offset[0]; break;
467 					// Y references
468 					case "tt": r = ctx.top + offset[1]; break;
469 					case "tb": r = ctx.bottom + offset[1]; break;
470 					case "bt": r = ctx.top - $element.outerHeight() + offset[1]; break;
471 					case "bb": r = ctx.bottom - $element.outerHeight() + offset[1]; break;
472 					case "mm": r = ctx.top + (ctx.height / 2) - ($element.outerHeight() / 2) + offset[1]; break;
473 					}
474 
475 					return r;
476 				},
477 
478 				// Splitted points
479 					splittedPoints = pts.split(" ");
480 
481 				// Calculates left and top with references to X and Y axis points (crossed points)
482 				return {
483 					"left": calculate(splittedPoints[0].charAt(0) + splittedPoints[1].charAt(0)),
484 					"top": calculate(splittedPoints[0].charAt(1) + splittedPoints[1].charAt(1))
