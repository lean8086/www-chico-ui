1 /**
  2 * Zoom shows a contextual reference to an augmented version of main declared image.
  3 * @name Zoom
  4 * @class Zoom
  5 * @augments ch.Floats
  6 * @requires ch.onImagesLoads
  7 * @memberOf ch
  8 * @param {Object} [conf] Object with configuration properties.
  9 * @param {Boolean} [conf.fx] Enable or disable fade effect on show. By default, the effect are disabled.
 10 * @param {Boolean} [conf.context] Sets a reference to position of component that will be considered to carry out the position. By default is the anchor of HTML snippet.
 11 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or "lt rt" by default.
 12 * @param {String} [conf.offset] Sets the offset in pixels that component will be displaced from original position determined by points. It's specified by configuration or "20 0" by default.
 13 * @param {String} [conf.content] This message will be shown when component needs to communicate that is in process of load. It's "Loading zoom..." by default.
 14 * @param {Number} [conf.width] Width of floated area of zoomed image. Example: 500, "500px", "50%". Default: 350.
 15 * @param {Number} [conf.height] Height of floated area of zoomed image. Example: 500, "500px", "50%". Default: 350.
 16 * @returns itself
 17 * @exampleDescription Create a Zoom component wrapping the original image with a anchor element pointing to a bigger version than the original.
 18 * @example
 19 * var widget = $(".example").zoom();
 20 * @factorized
 21 * @see ch.Floats
 22 * @see ch.Modal
 23 * @see ch.Tooltip
 24 * @see ch.Layer
 25 * @see ch.OnImagesLoads

242 		/**
243 		 * Reference to the augmented version of image, that will be displayed into a floated element.
244 		 * @private
245 		 * @name ch.Zoom#zoomed
246 		 * @type Object
247 		 */
248 			zoomed = (function () {
249 				// Define the content source
250 				var $img = that.source = $("<img src=\"" + that.element.href + "\">").insertAfter(original.$image);
251 
252 				// Grab some data when zoomed image loads
253 				$img.onImagesLoads(function () {
254 
255 					that.content.configure({
256 						'input': that.source
257 					});
258 
259 					// Save the zoomed image size
260 					zoomed.width = $img.prop("width");
261 					zoomed.height = $img.prop("height");
262 
263 					// Save the zoom ratio
264 					ratio.width = zoomed.width / original.width;
265 					ratio.height = zoomed.height / original.height;
266 
267 					// Seeker: Size relative to zoomed image respect zoomed area
268 					var w = ~~(conf.width / ratio.width),
269 						h = ~~(conf.height / ratio.height);
270 
271 					// Seeker: Save half width and half height
272 					seeker.width = w / 2;
273 					seeker.height = h / 2;
274 
275 					// Seeker: Set size and append it
276 					seeker.$shape.css({"width": w, "height": h}).appendTo(that.$element);
277 
278 					// Remove loading
279 					$loading.remove();
280 
281 					// Change zoomed image status to Ready
282 					zoomed.ready = true;
283 
284 					// TODO: MAGIC here! if mouse is over image show seeker and make all that innerShow do
285 				});

326 		/**
327 		 * Calculates movement limits and sets it to seeker and augmented image.
328 		 * @private
329 		 * @function
330 		 * @name ch.Zoom#move
331 		 * @param {Event} event Mouse event to take the cursor position.
332 		 */
333 			move = function (event) {
334 
335 				var x, y;
336 
337 				// Left side of seeker LESS THAN left side of image
338 				if (event.pageX - seeker.width < offset.left) {
339 					x = 0;
340 				// Right side of seeker GREATER THAN right side of image
341 				} else if (event.pageX + seeker.width > original.width + offset.left) {
342 					x = original.width - (seeker.width * 2) - 2;
343 				// Free move
344 				} else {
345 					x = event.pageX - offset.left - seeker.width;
346 				}
347 
348 				// Top side of seeker LESS THAN top side of image
349 				if (event.pageY - seeker.height < offset.top) {
350 					y = 0;
351 				// Bottom side of seeker GREATER THAN bottom side of image
352 				} else if (event.pageY + seeker.height > original.height + offset.top) {
353 					y = original.height - (seeker.height * 2) - 2;
354 				// Free move
355 				} else {
356 					y = event.pageY - offset.top - seeker.height;
357 				}
358 
359 				// Move seeker
360 				seeker.$shape.css({"left": x, "top": y});
361 
362 				// Move zoomed image
363 				zoomed.$image.css({"left": (-ratio.width * x), "top": (-ratio.height * y)});
364 
365 			};

371 		/**
372 		 * Inner show method. Attach the component's layout to the DOM tree and load defined content.
373 		 * @protected
374 		 * @name ch.Zoom#innerShow
375 		 * @function
376 		 * @returns itself
377 		 */
378 		that.innerShow = function () {
379 
380 			// If the component isn't loaded, show loading transition
381 			if (!zoomed.ready) {
382 				$loading.removeClass("ch-hide");
383 				return that;
384 			}
385 
386 			// Update position of anchor here because Zoom can be inside a Carousel and its position updates
387 			offset = that.$element.offset();
388 
389 			// Bind move calculations
390 			that.$element.bind("mousemove", function (event) { move(event); });
391 
392 			// Show seeker
393 			seeker.$shape.removeClass("ch-hide");
394 
395 			// Show float
396 			that.parent.innerShow();
397 
398 			return that;
399 
400 		};

402 		/**
403 		 * Inner hide method. Hides the component's layout and detach it from DOM tree.
404 		 * @protected
405 		 * @name ch.Zoom#innerHide
406 		 * @function
407 		 * @returns itself
408 		 */
409 		that.innerHide = function () {
410 
411 			// If the component isn't loaded, hide loading transition
412 			if (!zoomed.ready) {
413 				$loading.addClass("ch-hide");
414 				return that;
415 			}
416 
417 			// Unbind move calculations
418 			that.$element.unbind("mousemove");
419 
420 			// Hide seeker
421 			seeker.$shape.addClass("ch-hide");
422 
423 			// Hide float
424 			that.parent.innerHide();
425 
426 			return that;
427 
428 		};

430 		/**
431 		 * Getter and setter for size attributes of float that contains the zoomed image.
432 		 * @protected
433 		 * @function
434 		 * @name ch.Zoom#size
435 		 * @param {string} prop Property that will be setted or getted, like "width" or "height".
436 		 * @param {string} [data] Only for setter. It's the new value of defined property.
437 		 * @returns itself
438 		 */
439 		that.size = function (prop, data) {
440 
441 			// Seeker: Updates styles and size value
442 			if (data) {
443 				// Seeker: Size relative to zoomed image respect zoomed area
444 				var size = ~~(data / ratio[prop]);
445 
446 				// Seeker: Save half width and half height
447 				seeker[prop] = size / 2;
448 
449 				// Seeker: Set size
450 				seeker.$shape.css(prop, size);
451 			}
452 
453 			// Change float size
454 			return that.parent.size(prop, data);
455 		};
