1 /**
  2 * Tabs lets you create tabs for static and dynamic content.
  3 * @name Tabs
  4 * @class Tabs
  5 * @augments ch.Widget
  6 * @memberOf ch
  7 * @param {Object} [conf] Object with configuration properties.
  8 * @param {Number} [conf.selected] Selects a child that will be open when component was loaded. By default, the value is 1.
  9 * @returns itself
 10 * @factorized
 11 * @exampleDescription Create a new Tab Navigator without configuration.
 12 * @example
 13 * var widget = $(".example").tabs();
 14 * @exampleDescription Create a new Tab Navigator with configuration.
 15 * @example
 16 * var widget = $(".example").tabs({
 17 *     "selected": 2
 18 * });
 19 * @see ch.Widget
 20 */
 21 (function (window, $, ch) {
 22 	'use strict';
 23 
 24 	if (window.ch === undefined) {
 25 		throw new window.Error('Expected ch namespace defined.');
 26 	}

 28 	function Tabs($el, conf) {
 29 		/**
 30 		 * Reference to a internal component instance, saves all the information and configuration properties.
 31 		 * @private
 32 		 * @name ch.Tabs#that
 33 		 * @type object
 34 		 */
 35 		var that = this;
 36 
 37 		that.$element = $el;
 38 		that.element = $el[0];
 39 		that.type = 'tabs';
 40 		conf = conf || {};
 41 
 42 		conf = ch.util.clone(conf);
 43 
 44 		that.conf = conf;
 45 
 46 		/**
 47 		 * Inheritance
 48 		 */
 49 
 50 		that = ch.Widget.call(that);
 51 		that.parent = ch.util.clone(that);
 52 
 53 	/**
 54 	*	Private Members
 55 	*/
 56 
 57 		// Add CSS class to the main element
 58 		that.$element.addClass("ch-tabs");
 59 
 60 		/**
 61 		* The actual location hash, is used to know if there's a specific tab selected.
 62 		* @private
 63 		* @name ch.Tabs#hash
 64 		* @type string
 65 		*/
 66 		var hash = window.location.hash.replace("#!/", ""),
 67 
 68 		/**
 69 		* A boolean property to know if the some tag should be selected.
 70 		* @private
 71 		* @name ch.Tabs#hashed
 72 		* @type boolean
 73 		* @default false
 74 		*/
 75 			hashed = false,
 76 
 77 		/**
 78 		* Get wich tab is selected.
 79 		* @private
 80 		* @name ch.Tabs#selected
 81 		* @type number
 82 		*/
 83 			selected = conf.selected || conf.num || undefined,
 84 
 85 		/**
 86 		* Create controller's children.
 87 		* @private
 88 		* @name ch.Tabs#createTabs
 89 		* @function
 90 		*/
 91 			createTabs = function () {
 92 
 93 				// Children
 94 				that.$triggers.find("a").each(function (i, e) {
 95 
 96 					// Tab context
 97 					var tab = {};
 98 						tab.type = tab.name = "tab";
 99 						tab.element = e;
100 						tab.$element = $(e);
101 						tab.controller = that["public"];
102 
103 					// Tab configuration
104 					var config = {};
105 						config.open = (selected === i);
106 						config.onShow = function () { selected = i; };
107 
108 					if (ch.util.hasOwn(that.conf, "cache")) { config.cache = that.conf.cache; }
109 
110 					/**
111 					* Fired when the content of one dynamic tab loads.
112 					* @name ch.Tabs#contentLoad
113 					* @event
114 					* @public
115 					*/
116 					if (ch.util.hasOwn(that.conf, "onContentLoad")) { config.onContentLoad = that.conf.onContentLoad; }
117 
118 					/**
119 					* Fired when the content of one dynamic tab did not load.
120 					* @name ch.Tabs#contentError
121 					* @event
122 					* @public
123 					*/
124 					if (ch.util.hasOwn(that.conf, "onContentError")) { config.onContentError = that.conf.onContentError; }
125 
126 					// Create Tabs
127 					that.children.push(ch.Tab.call(tab, config));
128 
129 					// Bind new click to have control
130 					$(e).on("click", function (event) {
131 						ch.util.prevent(event);
132 						select(i);
133 					});
134 
135 				});
136 
137 				return;
138 
139 			},
140 
141 		/**
142 		* Select a child to show its content.
143 		* @name ch.Tabs#select
144 		* @private
145 		* @function
146 		*/
147 			select = function (index) {
148 
149 					// Sets the tab's index
150 				var tab = that.children[index];
151 
152 				// If select a tab that doesn't exist do nothing
153 				// Don't click me if I'm open
154 				if (!tab || index === selected) {
155 					return that;
156 				}
157 
158 				// Hides the open tab
159 				if (typeof selected !== "undefined") {
160 					that.children[selected].innerHide();
161 				}
162 
163 				// Shows the current tab
164 				tab.innerShow();
165 
166 				// Updated selected index
167 				selected = index;
168 
169 				//Change location hash
170 				window.location.hash = "#!/" + tab.$content.attr("id");
171 
172 				/**
173 				* Fired when a tab is selected.
174 				* @name ch.Tabs#select
175 				* @event
176 				* @public
177 				*/
178 				that.trigger("select");
179 
180 				// Callback
181 				that.callbacks("onSelect");
182 
183 				return that;
184 			};
185 
186 	/**
187 	*	Protected Members
188 	*/
189 
190 		/**
191 		* Collection of children.
192 		* @name ch.Form#children
193 		* @type {Array}
194 		*/
195 		that.children = [];
196 
197 
198 		/**
199 		* The component's triggers container.
200 		* @protected
201 		* @name ch.Tabs#$triggers
202 		* @type jQuery
203 		*/
204 		that.$triggers = that.$element.children(":first").addClass("ch-tabs-triggers").attr("role", "tablist");
205 
206 		/**
207 		* The component's content.
208 		* @protected
209 		* @name ch.Tabs#$content
210 		* @type jQuery
211 		*/
212 		that.$content = that.$triggers.next().addClass("ch-tabs-content ch-box-lite").attr("role", "presentation");
213 
214 	/**
215 	*	Public Members
216 	*/
217 
218 		/**
219 		 * @borrows ch.Widget#uid as ch.Tabs#uid
220 		 * @borrows ch.Widget#element as ch.Tabs#element
221 		 * @borrows ch.Widget#type as ch.Tabs#type
222 		 */
223 
224 		/**
225 		* Children instances associated to this controller.
226 		* @public
227 		* @name ch.Tabs#children
228 		* @type collection
229 		*/
230 		that["public"].children = that.children;
231 
232 		/**
233 		* Select a specific tab or get the selected tab.
234 		* @public
235 		* @name ch.Tabs#select
236 		* @function
237 		* @param {Number} [tab] Tab's index.
238 		* @exampleDescription Selects a specific tab
239 		* @example
240 		* widget.select(2);
241 		* @exampleDescription Returns the selected tab's index
242 		* @example
243 		* var selected = widget.select();
244 		*/
245 		that["public"].select = function (tab) {
246 			// Returns selected tab instead set it
247 			// Getter
248 			if (!parseInt(tab)) {
249 				return selected + 1;
250 			}
251 
252 			// Setter
253 			select(tab -= 1);
254 			return that["public"];
255 		};
256 
257 	/**
258 	*	Default event delegation
259 	*/
260 
261 		createTabs();
262 
263 		// If hash open that tab
264 		for(var i = that.children.length; i--;) {
265 			if (that.children[i].$content.attr("id") === hash) {
266 				select(i);
267 				hashed = true;
268 				break;
269 			}
270 		};
271 
272 		// Shows the first tab if not hash or it's hash and it isn't from the current tab
273 		if( !hash || ( hash && !hashed ) ){
274 			that.children[0].innerShow();
275 			selected = 0;
276 		}
277 
278 		/**
279 		* Triggers when the component is ready to use (Since 0.8.0).
280 		* @name ch.Tabs#ready
281 		* @event
282 		* @public
283 		* @since 0.8.0
284 		* @example
285 		* // Following the first example, using <code>widget</code> as Tabs's instance controller:
286 		* widget.on("ready",function () {
287 		*	this.show();
288 		* });
289 		*/
290 		//This avoit to trigger execute after the component was instanciated
291 		setTimeout(function(){that.trigger("ready")}, 50);
292 
293 		return that['public'];
294 	}

297 	Tabs.prototype.constructor = Tabs;
298

300 
301 	$.fn.tabNavigator = $.fn.tabs;

306 /**
307 * Tab is a simple unit of content for Tabs.
308 * @abstract
309 * @name Tab
310 * @class Tab
311 * @augments ch.Navs
312 * @memberOf ch
313 * @param {object} conf Object with configuration properties
314 * @returns itself
315 * @ignore
316 */
317 (function (window, $, ch) {
318 	'use strict';
319 
320 	if (window.ch === undefined) {
321 		throw new window.Error('Expected ch namespace defined.');
322 	}
323 
324 	function Tab(conf) {
325 		/**
326 		* Reference to a internal component instance, saves all the information and configuration properties.
327 		* @private
328 		* @name ch.Tab#that
329 		* @type object
330 		* @ignore
331 		*/
332 		var that = this;
333 		conf = ch.util.clone(conf);
334 		conf.icon = false;
335 
336 		that.conf = conf;
337 
338 	/**
339 	*	Inheritance
340 	*/
341 
342 		that = ch.Widget.call(that);
343 		that.parent = ch.util.clone(that);
344 
345 	/**
346 	 * Abilities
347 	 */
348 
349 		ch.Content.call(that);
350 
351 		/**
352 		 * This callback is triggered when content request have finished.
353 		 * @protected
354 		 * @name ch.Floats#onmessage
355 		 * @function
356 		 * @returns {this}
357 		 */
358 		that.content.onmessage = function (data) {
359 
360 			that.$content.html(data);
361 
362 			that.controller.trigger("contentLoad");
363 
364 			if (ch.util.hasOwn(conf, "onContentLoad")) {
365 				conf.onContentLoad.call(that.controller, data);
366 			}
367 		};
368 
369 		/**
370 		 * This callback is triggered when async request fails.
371 		 * @protected
372 		 * @name ch.Floats#onerror
373 		 * @function
374 		 * @returns {this}
375 		 */
376 		that.content.onerror = function (data) {
377 
378 			that.$content.html(data);
379 
380 			that.controller.trigger("contentError", data);
381 
382 			if (ch.util.hasOwn(conf, "onContentError")) {
383 				conf.onContentError.call(that.controller, data.jqXHR, data.textStatus, data.errorThrown);
384 			}
385 		};
386 
387 	/**
388 	*	Private Members
389 	*/
390 		/**
391 		* Creates the basic structure for the tab's content.
392 		* @private
393 		* @name ch.Tab#createContent
394 		* @function
395 		* @ignore
396 		*/
397 		var createContent = function () {
398 
399 			var href = that.element.href.split("#"),
400 				controller = that.$element.parents(".ch-tabs"),
401 				content = controller.find("#" + href[1]);
402 
403 			// If there are a tabContent...
404 			if (content.length > 0) {
405 				return content;
406 
407 			// If tabContent doesn't exists
408 			} else {
409 				/**
410 				* Content configuration property.
411 				* @public
412 				* @name ch.Tab#source
413 				* @type string
414 				* @ignore
415 				*/
416 				that.source = that.element.href;
417 
418 				that.content.configure({
419 					'input': that.source
420 				});
421 
422 				var id = (href.length === 2) ? href[1] : "ch-tab-" + that.uid;
423 
424 				// Create tabContent
425 				return $("<div id=\"" + id + "\" role=\"tabpanel\" class=\"ch-hide\">").appendTo(controller.children().eq(1));
426 			}
427 
428 		};
429 
430 	/**
431 	*	Protected Members
432 	*/
433 		/**
434 		* Reference to the trigger element.
435 		* @protected
436 		* @name ch.Tab#$trigger
437 		* @type jQuery
438 		* @ignore
439 		*/
440 		that.$trigger = that.$element;
441 
442 		/**
443 		* The component's content.
444 		* @protected
445 		* @name ch.Tab#$content
446 		* @type jQuery
447 		* @ignore
448 		*/
449 		that.$content = createContent();
450 
451 		/**
452 		* Shows component's content.
453 		* @protected
454 		* @function
455 		* @name ch.Tab#innerShow
456 		* @returns itself
457 		* @ignore
458 		*/
459 		that.innerShow = function (event) {
460 			ch.util.prevent(event);
461 
462 			that.active = true;
463 
464 			// Load my content if I'need an ajax request
465 			if (ch.util.hasOwn(that, "source")) { that.content.set(); }
466 
467 			// Show me
468 			that.$trigger.addClass("ch-" + that["type"] + "-trigger-on");
469 
470 			// Set me as hidden false
471 			that.$content
472 				.attr("aria-hidden", "false")
473 				.removeClass("ch-hide");
474 
475 			return that;
476 		};
477 
478 		/**
479 		* Hides component's content.
480 		* @protected
481 		* @function
482 		* @name ch.Tab#innerHide
483 		* @returns itself
484 		* @ignore
485 		*/
486 		that.innerHide = function (event) {
487 			ch.util.prevent(event);
488 
489 			if (!that.active) { return; }
490 
491 			that.active = false;
492 
493 			// Hide me
494 			that.$trigger.removeClass("ch-" + that["type"] + "-trigger-on");
495 
496 			// Set all inactive tabs as hidden
497 			that.$content
498 				.attr("aria-hidden", "true")
499 				.addClass("ch-hide");
500 
501 			return that;
502 		};
503 
504 
505 	/**
506 	*	Public Members
507 	*/
508 
509 	/**
510 	*	Default event delegation
511 	*/
512 
513 		// Add the attributes for WAI-ARIA to the tabs and tabpanel
514 		// By default is hidden
515 		that.$content.attr({
516 			"role": "tabpanel",
517 			"aria-hidden": true,
518 			"class": "ch-hide"
519 		});
520 
521 		that.$trigger.attr({
522 			"role": "tab",
523 			"arial-controls": that.$content.attr("id"),
524 			"class": "ch-tab-trigger"
525 		});
526 
527 		return that;
528 	}

530 	Tab.prototype.name = 'tab';
531 	Tab.prototype.constructor = Tab;
532 
533 	ch.Tab = Tab;
534 
535 }(this, this.jQuery, this.ch));
