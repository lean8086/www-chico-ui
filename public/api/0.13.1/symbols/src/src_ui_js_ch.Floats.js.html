 63 		that.content.onmessage = function (data) {
 64 
 65 			if (!that.active) { return; }
 66

 72 			}
 73 
 74 			that.position('refresh');

 84 		that.content.onerror = function (data) {
 85 
 86 			if (!that.active) { return; }
 87 
 88 			that.$content.html(data);
 89 
 90 			that.trigger("contentError");
 91 			if (ch.util.hasOwn(conf, "onContentError")) {
 92 				conf.onContentError.call((that.controller || that), data.jqXHR, data.textStatus, data.errorThrown);
 93 			}
 94 
 95 			that.position('refresh');
 96 		};

120 		 * @private
121 		 * @function
122 		 * @name ch.Floats-closable
123 		 */
124 		// TODO: Create "closable" interface
125 		var closable = (function () {
126 			/**
127 			 * Returns any if the component closes automatic.
128 			 * @public
129 			 * @function
130 			 * @methodOf ch.Floats#closabe
131 			 * @exampleDescription to get the height
132 			 * @example
133 			 * widget.closable() // true | false | "button"
134 			 * @returns boolean | string
135 			 */
136 			that["public"].closable = function () {
137 				return that.closable;
138 			};
139 
140 
141 			return function () {
142 
143 				// Closable Off: don't anything
144 				if (!that.closable) { return; }
145 
146 				// Closable On
147 				if (ch.util.hasOwn(conf, "closeButton") && conf.closeButton || ch.util.hasOwn(conf, "event") && conf.event === "click") {
148 					// Append close buttons
149 					// It will close with close button
150 					that.$container
151 						.prepend("<a class=\"ch-close\" role=\"button\" style=\"z-index:" + (ch.util.zIndex += 1) + "\"></a>")
152 						.on("click", function (event) {
153 							if ($(event.target || event.srcElement).hasClass("ch-close")) {
154 								that.innerHide(event);
155 							}
156 						});
157 				}
158 
159 				// ESC key support
160 				$document.on(ch.events.key.ESC, function () {
161 					that.innerHide();
162 				});
163 
164 				// It will close only with close button
165 				if (that.closable === "button") {
166 					return;
167 				}
168 
169 				// Default Closable behavior
170 				// It will close with click on document, too
171 				that.on("show", function () {
172 					$document.one("click", that.innerHide);
173 				});
174 
175 				// Stop event propatation, if click container.
176 				that.$container.on("click", function (event) {
177 					event.stopPropagation();
178 				});
179 			};

203 		/**
204 		 * Inner function that resolves the component's layout and returns a static reference.
205 		 * @protected
206 		 * @name ch.Floats#$container
207 		 * @type jQuery
208 		 */
209 		that.$container = (function () {
210 
211 			// Final jQuery Object
212 			var $container,
213 
214 			// HTML Div Element with role for WAI-ARIA
215 			container = ["<div role=\"" + conf.aria.role + "\""];
216 
217 			var id = "ch-" + that.type + "-" + that.uid;
218 			// ID for WAI-ARIA
219 			if (ch.util.hasOwn(conf.aria, "identifier")) {
220 				// Generated ID using component name and its instance order
221 
222 				// Add aria attribute to trigger element
223 				that.$element.attr(conf.aria.identifier, id);
224 			}
225 
226 			// Add ID to container element
227 			container.push(" id=\"" + id + "\"");
228 
229 			// Classname with component type and extra classes from conf.classes
230 			container.push(" class=\"ch-hide ch-" + that.type + (ch.util.hasOwn(conf, "classes") ? " " + conf.classes : "") + "\"");
231 
232 			// Z-index
233 			container.push(" style=\"z-index:" + (ch.util.zIndex += 1) + ";");
234 
235 			// Width
236 			if (ch.util.hasOwn(conf, "width")) {
237 				container.push("width:" + conf.width + ((typeof conf.width === "number") ? "px;" : ";"));
238 			}
239 
240 			// Height
241 			if (ch.util.hasOwn(conf, "height")) {
242 				container.push("height:" + conf.height + ((typeof conf.height === "number") ? "px;" : ";"));
243 			}
244 
245 			// Style and tag close
246 			container.push("\">");
247 
248 			// Create cone
249 			if ($html.hasClass("lt-ie8") && ch.util.hasOwn(conf, "cone")) {
250 				container.push("<div class=\"ch-" + that.type + "-cone\"></div>");
251 			}
252 
253 			// Tag close
254 			container.push("</div>");
255 
256 			// jQuery Object generated from string
257 			$container = $(container.join(""));
258 
259 			// Create cone
260 			if (ch.util.hasOwn(conf, "cone")) { $container.addClass("ch-cone"); }
261 
262 			// Efects configuration
263 			conf.fx = ch.util.hasOwn(conf, "fx") ? conf.fx : true;
264 
265 			// Position component configuration
266 			conf.position = conf.position || {};
267 			conf.position.element = $container;
268 			conf.position.reposition = ch.util.hasOwn(conf, "reposition") ? conf.reposition : true;
269 
270 			// Initialize positioner component
271 			that.position = ch.Positioner(conf.position);
272 
273 			// Return the entire Layout
274 			return $container;
275 		})();

286 		/**
287 		 * Inner show method. Attach the component layout to the DOM tree.
288 		 * @protected
289 		 * @function
290 		 * @name ch.Floats#innerShow
291 		 * @returns itself
292 		 */
293 		that.innerShow = function (event) {
294 			if (event) {
295 				ch.util.prevent(event);
296 			}
297 
298 			// Avoid showing things that are already shown
299 			if (that.active) return;
300 
301 			that.active = true;
302 
303 			// Add layout to DOM tree
304 			// Increment zIndex
305 			that.$container
306 				.appendTo("body")
307 				.css("z-index", ch.util.zIndex++);
308 
309 			// Request the content
310 			that.content.set();
311 
312 			/**
313 			 * Triggers when component is visible.
314 			 * @name ch.Floats#show
315 			 * @event
316 			 * @public
317 			 * @exampleDescription It change the content when the component was shown.
318 			 * @example
319 			 * widget.on("show",function () {
320 			 * this.content("Some new content");
321 			 * });
322 			 * @see ch.Floats#show
323 			 */
324 			// Show component with effects
325 			if (conf.fx) {
326 				that.$container.fadeIn("fast", function () {
327 
328 					that.$container.removeClass("ch-hide");
329 					// new callbacks
330 					that.trigger("show");
331 					// Old callback system
332 					that.callbacks('onShow');
333 
334 				});
335 			} else {
336 			// Show component without effects
337 				that.$container.removeClass("ch-hide");
338 				// new callbacks
339 				that.trigger("show");
340 				// Old callback system
341 				that.callbacks('onShow');
342 			}
343 
344 			that.position("refresh");
345 
346 			return that;
347 		};

349 		/**
350 		 * Inner hide method. Hides the component and detach it from DOM tree.
351 		 * @protected
352 		 * @function
353 		 * @name ch.Floats#innerHide
354 		 * @returns itself
355 		 */
356 		that.innerHide = function (event) {
357 
358 			if (event) {
359 				event.stopPropagation();
360 			}
361 
362 			if (!that.active) { return; }
363 
364 			that.active = false;
365 
366 			var afterHide = function () {
367 				/**
368 				 * Triggers when component is not longer visible.
369 				 * @name ch.Floats#hide
370 				 * @event
371 				 * @public
372 				 * @exampleDescription When the component hides show other component.
373 				 * @example
374 				 * widget.on("hide",function () {
375 				 * otherComponent.show();
376 				 * });
377 				 */
378 				// new callbacks
379 				that.trigger("hide");
380 				// Old callback system
381 				that.callbacks('onHide');
382 
383 				that.$container.detach();
384 
385 			};
386 
387 			// Show component with effects
388 			if (conf.fx) {
389 				that.$container.fadeOut("fast", afterHide);
390 
391 			// Show component without effects
392 			} else {
393 				that.$container.addClass("ch-hide");
394 				afterHide();
395 			}
396 
397 			// Removes the innerHide listener
398 			// #708 Modal: The widget closes by itself when It's showing the second time
399 			$document.off("click", that.innerHide);
400 
401 			return that;
402 
403 		};

409 		 * @name ch.Floats#size
410 		 * @param {String} prop Property that will be setted or getted, like "width" or "height".

414 		that.size = function (prop, data) {
415 			// Getter
416 			if (!data) { return that.conf[prop]; }
417 
418 			// Setter
419 			that.conf[prop] = data;
420 
421 			// Container size
422 			that.$container[prop](data);
423 
424 			// Refresh position
425 			that.position("refresh");
426 
427 			return that["public"];
428 		};

491 		/**
492 		 * Sets or gets positioning configuration. Use it without arguments to get actual configuration. Pass an argument to define a new positioning configuration.
493 		 * @public
494 		 * @function
495 		 * @name ch.Floats#position
496 		 * @exampleDescription Change component's position.
497 		 * @example
498 		 * widget.position({
499 		 *   offset: "0 10",
500 		 *   points: "lt lb"
501 		 * });
502 		 * @exampleDescription Refresh position.
503 		 * @example
504 		 * widget.position("refresh");
505 		 * @see ch.Floats#position
506 		 */
507 		// Create a custom Positioner object to update conf.position data of Float family
508 		that["public"].position = function (o) {
509 
510 			var r = that["public"];
511 
512 			switch (typeof o) {
513 			// Custom Setter: It updates conf.position data
514 			case "object":
515 				// New points
516 				if (ch.util.hasOwn(o, "points")) { conf.position.points = o.points; }
517 
518 				// New reposition
519 				if (ch.util.hasOwn(o, "reposition")) { conf.position.reposition = o.reposition; }
520 
521 				// New offset (splitted)
522 				if (ch.util.hasOwn(o, "offset")) { conf.position.offset = o.offset; }
523 
524 				// New context
525 				if (ch.util.hasOwn(o, "context")) { conf.position.context = o.context; }
526 
527 				// Original Positioner
528 				that.position(conf.position);
529 
530 				break;
531 
532 			// Refresh
533 			case "string":
534 				that.position("refresh");
535 
536 				break;
537 
538 			// Getter
539 			case "undefined":
540 			default:
541 				r = that.position();
542 
543 				break;
544 			}
545 
546 			return r;
547 		};
