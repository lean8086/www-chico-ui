256 		/**
257 		* Indicates if an specific date is selected or not (including date ranges and simple dates).
258 		* @private
259 		* @name ch.Calendar#isSelectable
260 		* @function
261 		* @param year
262 		* @param month
263 		* @param day
264 		* @return Boolean
265 		*/
266 			isSelectable = function (year, month, day) {
267 
268 				if (!selected) { return; }
269 
270 				var yepnope = false;
271 
272 				// Simple selection
273 				if (!ch.util.isArray(selected)) {
274 					if (year === selected.year && month === selected.month && day === selected.day) {
275 						return yepnope = true;
276 					}
277 				// Multiple selection (ranges)
278 				} else {
279 					$.each(selected, function (i, e) {
280 						// Simple date
281 						if (!ch.util.isArray(e)) {
282 							if (year === e.year && month === e.month && day === e.day) {
283 								return yepnope = true;
284 							}
285 						// Range
286 						} else {
287 							if (
288 								(year >= e[0].year && month >= e[0].month && day >= e[0].day) &&
289 								(year <= e[1].year && month <= e[1].month && day <= e[1].day)
290 							) {
291 								return yepnope = true;
292 							}
293 						}
294 					});
295 				}
296 
297 				return yepnope;

324 		/**
325 		* Creates a complete month in a table.
326 		* @private
327 		* @function
328 		* @name ch.Calendar#createTable
329 		* @param date {Object} Date from will be created the entire month.
330 		* @return jQuery Object
331 		*/
332 			createTable = function (date) {
333 				// Total amount of days into month
334 				var cells = (function () {
335 
336 					// Amount of days of current month
337 					var currentMonth = new Date(date.year, date.month, 0).getDate(),
338 
339 					// Amount of days of previous month
340 						prevMonth = new Date([date.year, date.month, "01"].join("/")).getDay(),
341 
342 					// Merge amount of previous and current month
343 						subtotal = prevMonth + currentMonth,
344 
345 					// Amount of days into last week of month
346 						latest = subtotal % 7,
347 
348 					// Amount of days of next month
349 						nextMonth = (latest > 0) ? 7 - latest : 0;
350 
351 					return {
352 						"previous": prevMonth,
353 						"subtotal": subtotal,
354 						"total": subtotal + nextMonth
355 					};
356 
357 				}()),
358 
359 				// Final array with month table structure
360 					r = [
361 						"<table class=\"ch-calendar-month\" role=\"grid\" id=\"ch-calendar-grid-" + that.uid + "\">",
362 						"<caption>" + MONTHS_NAMES[date.month - 1] + " - " + date.year + "</caption>",
363 						thead,
364 						"<tbody>",
365 						"<tr class=\"ch-week\" role=\"row\">"
366 					];
367 
368 				// Iteration of weekdays
369 				for (var i = 0; i < cells.total; i += 1) {
370 
371 					// Push an empty cell on previous and next month
372 					if (i < cells.previous || i > cells.subtotal - 1) {
373 						r.push("<td role=\"gridcell\" class=\"ch-calendar-other\">X</td>");
374 					} else {
375 
376 						// Positive number of iteration
377 						var positive = i + 1,
378 
379 						// Day number
380 							day = positive - cells.previous,
381 
382 						// Define if it's the day selected
383 							isSelected = isSelectable(date.year, date.month, day);
384 
385 						// Create cell
386 						r.push(
387 							// Open cell structure including WAI-ARIA and classnames space opening
388 							"<td role=\"gridcell\"" + (isSelected ? " aria-selected=\"true\"" : "") + " class=\"ch-calendar-day",
389 
390 							// Add Today classname if it's necesary
391 							(date.year === today.year && date.month === today.month && day === today.day) ? " ch-calendar-today" : null,
392 
393 							// Add Selected classname if it's necesary
394 							(isSelected ? " ch-calendar-selected" : null),
395 
396 							// From/to range. Disabling cells
397 							(
398 								// Disable cell if it's out of FROM range
399 								(from && day < from.day && date.month === from.month && date.year === from.year) ||
400 
401 								// Disable cell if it's out of TO range
402 								(to && day > to.day && date.month === to.month && date.year === to.year)
403 
404 							) ? " ch-calendar-disabled" : null,
405 
406 							// Close classnames attribute and print content closing cell structure
407 							"\">" + day + "</td>"
408 						);
409 
410 						// Cut week if there are seven days
411 						if (positive % 7 === 0) {
412 							r.push("</tr><tr class=\"ch-calendar-week\" role=\"row\">");
413 						}
414 
415 					}
416 
417 				};
418 
419 				// Return table object
420 				return r.join("");
421 
422 			},
