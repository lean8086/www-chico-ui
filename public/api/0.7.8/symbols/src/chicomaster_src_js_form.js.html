  2 /**
  3 * Forms is a Controller of DOM's HTMLFormElement.
  4 * @name Form
  5 * @class Form
  6 * @augments ch.Controllers
  7 * @memberOf ch
  8 * @param {object} conf Object with configuration properties
  9 * @returns itself
 10 */
 11 
 12 ch.form = function(conf) {
 13 
 14 /**
 15 * Validation
 16 */
 17 	// Are there action and submit type?
 18 	if ( this.$element.find(":submit").length == 0 || this.$element.attr("action") == "" ){
 19 		alert("Form fatal error: The <input type=submit> is missing, or need to define a action attribute on the form tag.");
 20 		return;
 21 	};
 22 
 23 	// Is there form in map instances?
 24 	if ( ch.utils.hasOwn(ch.instances, "form") && ch.instances.form.length > 0 ){
 25 		for(var i = 0, j = ch.instances.form.length; i < j; i+=1){
 26 			if(ch.instances.form[i].element === this.element){
 27 				return {
 28 					exists: true,
 29 					object: ch.instances.form[i]
 30 				};
 31 			}
 32 		};
 33 	}
 34 
 35 	/**
 36 	* Reference to a internal component instance, saves all the information and configuration properties.
 37 	* @private
 38 	* @name ch.Form#that
 39 	* @type object
 40 	*/
 41 	var that = this;
 42 
 43 	conf = ch.clon(conf);
 44 	// Disable HTML5 browser-native validations
 45 	that.$element.attr("novalidate", "novalidate");
 46 	// Grab submit button
 47 	that.$submit = that.$element.find("input:submit");
 48 
 49 	that.conf = conf;
 50 
 51 /**
 52 *  Inheritance
 53 */
 54 
 55 	that = ch.controllers.call(that);
 56 	that.parent = ch.clon(that);
 57 	
 58 /**
 59 *  Private Members
 60 */
 61 
 62 	/**
 63 	* A Boolean property that indicates is exists errors in the form.
 64 	* @private
 65 	* @name ch.Form#status
 66 	* @type boolean
 67 	*/
 68 	var status = true;
 69 
 70 	/**
 71 	* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
 72 	*/
 73 	var validate = function(event){
 74 
 75 		/**
 76 		* Callback function
 77 		* @name ch.Form#beforeValidate
 78 		* @event
 79 		* @public
 80 		*/
 81 		that.callbacks("beforeValidate");
 82 		// new callback
 83 		that.trigger("beforeValidate");
 84 
 85 		// Status OK (with previous error)
 86 		if ( !status ) {
 87 			status = true;
 88 		};
 89 
 90 		var i = 0, j = that.children.length, toFocus, childrenError = [];
 91 
 92 		// Shoot validations
 93 		for (i; i < j; i+=1) {
 94 			var child = that.children[i];
 95 
 96 			// Validate
 97 			// Save children with errors
 98 			if ( child.hasError() ) {
 99 				childrenError.push(child);
100 			}
101 		};
102 
103 		// Is there's an error
104 		if (childrenError.length > 0) {
105 			status = false;
106 			// Issue UI-332: On validation must focus the first field with errors.
107 			// Doc: http://wiki.ml.com/display/ux/Mensajes+de+error
108 			if (childrenError[0].element.tagName === "DIV") {
109 				$(childrenError[0].element).find("input:first").focus();
110 			} else {
111 				childrenError[0].element.focus();
112 			}
113 		} else {
114 			status = true;
115 		}
116 
117 		/**
118 		* Callback function
119 		* @name ch.Form#validate
120 		* @event
121 		* @public
122 		*/
123 		/**
124 		* Callback function
125 		* @name ch.Form#error
126 		* @event
127 		* @public
128 		*/
129 		if (status) {
130 			that.callbacks("onValidate");
131 			// new callback
132 			that.trigger("validate");
133 		} else {
134 			that.callbacks("onError");
135 			// new callback
136 			that.trigger("error");
137 		}
138 
139 		/**
140 		* Callback function
141 		* @name ch.Form#afterValidate
142 		* @event
143 		* @public
144 		*/
145 		that.callbacks("afterValidate");
146 		// new callback
147 		that.trigger("afterValidate");
148 
149 		return that;
150 	};
151 
152 	/**
153 	* This methods triggers the 'beforSubmit' callback, then will execute validate() method,
154 	* and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
155 	*/
156 	var submit = function(event) {
157 
158 		/**
159 		* Callback function
160 		* @name ch.Form#beforeSubmit
161 		* @event
162 		* @public
163 		*/
164 		that.callbacks("beforeSubmit");
165 		// new callback
166 		that.trigger("beforeSubmit");
167 
168 		// re-asign submit event
169 		that.$element.one("submit", submit);
170 
171 		// Execute all validations
172 		validate(event);
173 
174 		// If an error ocurs prevent default actions
175 		if (!status) {
176 			that.prevent(event);
177 			event.stopImmediatePropagation();
178 		}
179 
180 		// OLD CALLBACK SYSTEM!
181 		// Is there's no error but there's a onSubmit callback
182 		if ( status && ch.utils.hasOwn(conf, "onSubmit")) {
183 			// Avoid default actions
184 			that.prevent(event);
185 			// To execute defined onSubmit callback
186 			that.callbacks("onSubmit");
187 		}
188 
189 		// OLD CALLBACK SYSTEM!
190 		// Is there's no error but there's a onSubmit callback
191 		if ( status && ch.utils.hasOwn(conf, "onSubmit")) {
192 			// Avoid default actions
193 			that.prevent(event);
194 			// To execute defined onSubmit callback
195 			that.callbacks("onSubmit");
196 		}
197 		/**
198 		* Callback function
199 		* @name ch.Form#submit
200 		* @event
201 		* @public
202 		*/
203 		// * New callback system *
204 		// Check inside $.data if there's a handler for ch-submit event
205 		// if something found there, avoid submit.
206 
207 		var formEvents = $(that["public"]).data("events");
208 		var isSubmitEventDefined = (formEvents && ch.utils.hasOwn(formEvents, "ch-submit"));
209 
210 		if (status && isSubmitEventDefined){
211 			// Avoid default actions
212 			that.prevent(event);
213 			// new callback
214 			that.trigger("submit");
215 		};
216 
217 		/**
218 		* Callback function
219 		* @name ch.Form#afterSubmit
220 		* @event
221 		* @public
222 		*/
223 		that.callbacks("afterSubmit");
224 		// new callback
225 		that.trigger("afterSubmit");
226 
227 		// Return that to chain methods
228 		return that;
229 	};
230 
231 	/**
232 	* Use this method to clear al validations.
233 	*/
234 	var clear = function(){
235 
236 		var i = 0, j = that.children.length;
237 		for(i; i < j; i += 1) {
238 		  that.children[i].clear();
239 		}
240 		
241 		status = true;
242 
243 		status = true;
244 
245 		/**
246 		* Callback function
247 		* @name ch.Form#onClear
248 		* @event
249 		* @public
250 		*/
251 		that.callbacks("onClear");
252 		// new callback
253 		that.trigger("clear");
254 
255 		return that;
256 	};
257 
258 	/**
259 	* Use this method to reset the form's input elements.
260 	*/
261 	var reset = function(event){
262 		clear();
263 		that.element.reset(); // Reset html form native
264 
265 		/**
266 		* Callback function
267 		* @name ch.Form#onReset
268 		* @event
269 		* @public
270 		*/
271 		that.callbacks("onReset");
272 		// new callback
273 		that.trigger("reset");
274 
275 		return that;
276 	};

304 	* Watcher instances associated to this controller.
305 	* @public
306 	* @name ch.Form#children
307 	* @type collection
308 	*/
309 	that["public"].children = that.children;
310

312 	* Collection of messages defined.
313 	* @public
314 	* @name ch.Form#messages
315 	* @type string
316 	*/
317 	that["public"].messages = conf.messages || {};
318 
319 	/**
320 	* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
321 	* @public
322 	* @function
323 	* @name ch.Form#validate
324 	* @returns itself
325 	*/
326 	that["public"].validate = function() {
327 		validate();
328 
329 		return that["public"];
330 	};
331 
332 	/**
333 	* This methods triggers the 'beforSubmit' callback, then will execute validate() method, and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
334 	* @public
335 	* @function
336 	* @name ch.Form#submit
337 	* @returns itself
338 	*/
339 	that["public"].submit = function() {
340 		submit();
341 
342 		return that["public"];
343 	};
344 
345 	/**
346 	* Return the status value.
347 	* @public
348 	* @function
349 	* @name ch.Form#getStatus
350 	* @returns itself
351 	*/
352 	that["public"].getStatus = function(){
353 		return status;
354 	};
355 
356 	/**
357 	* Use this method to clear al validations.
358 	* @public
359 	* @function
360 	* @name ch.Form#clear
361 	* @returns itself
362 	*/
363 	that["public"].clear = function() {
364 		clear();
365 
366 		return that["public"];
367 	};
368 
369 	/**
370 	* Use this method to clear al validations.
371 	* @public
372 	* @function
373 	* @name ch.Form#reset
374 	* @returns itself
375 	*/
376 	that["public"].reset = function() {
377 		reset();
378 
379 		return that["public"];
380 	};
381 
382 /**
383 *  Default event delegation
384 */
385 
386 	// patch exists because the components need a trigger
387 	if (ch.utils.hasOwn(conf, "onSubmit")) {
388 		that.$element.bind('submit', function(event){ that.prevent(event); });
389 		// Delete all click handlers asociated to submit button >NATAN: Why?
390 			//Because if you want do something on submit, you need that the trigger (submit button)
391 			//don't have events associates. You can add funcionality on onSubmit callback
392 		that.$element.find(":submit").unbind('click');
393 	};
394 
395 	// Bind the submit
396 	that.$element.bind("submit", function(event){ submit(event) });
397 
398 	// Bind the reset
399 	that.$element.find(":reset, .resetForm").bind("click", function(event){ reset(event); });
400 
401 	return that;
402 };
