  3 * Zoom is a standalone UI component that shows a contextual reference to an augmented version of main declared image.
  4 * @name Zoom

 14 ch.zoom = function(conf) {
 15 	/**
 16 	* Reference to an internal component instance, saves all the information and configuration properties.
 17 	* @private
 18 	* @name ch.Zoom#that
 19 	* @type itself
 20 	*/
 21 	var that = this;
 22 
 23 /**
 24 *	Constructor
 25 */
 26 	conf = ch.clon(conf);
 27 	conf.fx = false;
 28 	
 29 	conf.position = {};
 30 	conf.position.context = conf.context || that.$element;
 31 	conf.position.offset = conf.offset || "20 0";
 32 	conf.position.points = conf.points || "lt rt";
 33 	conf.position.hold = true;
 34 
 35 	that.conf = conf;
 36 
 37 /**
 38 *	Inheritance
 39 */
 40 
 41 	that = ch.floats.call(that);
 42 	that.parent = ch.clon(that);

 48 	/**
 49 	* Reference to main image declared on HTML code snippet.
 50 	* @private
 51 	* @name ch.Zoom#original
 52 	* @type object
 53 	*/
 54 	var original = {};
 55 		original.img = that.$element.children();
 56 		original["width"] = original.img.prop("width");
 57 		original["height"] = original.img.prop("height");
 58 
 59 	/**
 60 	* Reference to the augmented version of image, that will be displayed in context.
 61 	* @private
 62 	* @name ch.Zoom#zoomed
 63 	* @type object
 64 	*/
 65 	var zoomed = {};
 66 		// Define the content source 
 67 		zoomed.img = that.source = $("<img>").prop("src", that.element.href);
 68 	
 69 	/**
 70 	* Seeker is the visual element that follows mouse movement for referencing to zoomable area into original image.
 71 	* @private
 72 	* @name ch.Zoom#seeker
 73 	* @type object
 74 	*/
 75 	var seeker = {};
 76 		seeker.shape = $("<div>").addClass("ch-seeker ch-hide")
 77 	
 78 	/**
 79 	* Gets the mouse position relative to original image position, and accordingly moves the zoomed image.
 80 	* @private
 81 	* @function
 82 	* @name ch.Zoom#move
 83 	* @param event event
 84 	*/
 85 	var move = function(event){
 86 		
 87 		// Cursor coordinates relatives to original image
 88 		var x = event.pageX - original.offset.left;
 89 		var y = event.pageY - original.offset.top;
 90 		
 91 		// Seeker axis
 92 		var limit = {};
 93 			limit.left = parseInt(x - seeker["width"]);
 94 			limit.right = parseInt(x + seeker["width"]);
 95 			limit.top = parseInt(y - seeker["height"]);
 96 			limit.bottom = parseInt(y + seeker["height"]);
 97 
 98 		// Horizontal: keep seeker into limits
 99 		if(limit.left >= 0 && limit.right < original["width"] - 1) {
100 			zoomed.img.css("left", -( (parseInt(zoomed["width"]* x) / original["width"]) - (conf.width / 2) ));
101 			seeker.shape.css("left", limit.left);
102 		};
103 		
104 		// Vertical: keep seeker into limits
105 		if(limit.top >= 0 && limit.bottom < original["height"] - 1) {
106 			zoomed.img.css("top", -( (parseInt(zoomed["height"]* y) / original["height"]) - (conf.height / 2) ));
107 			seeker.shape.css("top", limit.top);
108 		};
109 		
110 	};
111 
112 	/**
113 	* Calculates zoomed image sizes and adds event listeners to trigger of float element
114 	* @private
115 	* @function
116 	* @name ch.Zoom#init
117 	*/
118 	var init = function(){
119 		// Zoomed image size
120 		zoomed["width"] = zoomed.img.prop("width");
121 		zoomed["height"] = zoomed.img.prop("height");
122 		
123 		// Anchor
124 		that.$element
125 			// Apend Seeker
126 			.append( seeker.shape )
127 			
128 			// Show
129 			.bind("mouseenter", that.show)
130 			
131 			// Hide
132 			.bind("mouseleave", that.hide)
133 	};

135 /**
136 *	Protected Members

138 
139 	/**
140 	* Anchor that wraps the main image and links to zoomed image file.
141 	* @protected
142 	* @name ch.Zoom#$trigger
143 	* @type jQuery
144 	*/
145 	that.$trigger = that.$element;
146 
147 	that.innerShow = function(){
148 		// Recalc offset of original image
149 		original.offset = original.img.offset();
150 
151 		// Move
152 		that.$element.bind("mousemove", function(event){ 
153 			move(event); 
154 		});
155 
156 		// Seeker
157 		seeker.shape.removeClass("ch-hide");
158 		
159 		// Floats show
160 		that.parent.innerShow();
161 
162 		return that;
163 	};
164 
165 	that.innerHide = function(){
166 		// Move
167 		that.$element.unbind("mousemove");
168 		
169 		// Seeker
170 		seeker.shape.addClass("ch-hide");
171 		
172 		// Floats hide
173 		that.parent.innerHide();
174 		
175 		return that;
176 	};

178 	/**
179 	* Triggered on anchor click, it prevents redirection to zoomed image file.
180 	* @protected
181 	* @function
182 	* @name ch.Zoom#enlarge
183 	* @param {mouseEvent} event
184 	* @returns itself
185 	*/
186 	that.enlarge = function(event){
187 		that.prevent(event);
188 		// Do what you want...
189 		return that;
190 	};

192 	/**
193 	* Getter and setter for size attributes of float that contains the zoomed image.
194 	* @protected
195 	* @function
196 	* @name ch.Zoom#size
197 	* @param {string} prop Property that will be setted or getted, like "width" or "height".
198 	* @param {string} [data] Only for setter. It's the new value of defined property.
199 	* @returns itself
200 	*/
201 	that.size = function(prop, data) {
202 
203 		if (data) {
204 
205 			// Seeker: shape size relative to zoomed image respect zoomed area
206 			var size = (original[prop]* data) / zoomed[prop];
207 
208 			// Seeker: sets shape size
209 			seeker.shape[prop](size);
210 
211 			// Seeker: save shape half size for position it respect cursor
212 			seeker[prop] = size / 2;
213 
214 		};
215 
216 		return that.parent.size(prop, data);
217 	};
