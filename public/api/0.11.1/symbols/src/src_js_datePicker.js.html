1 /**
  2 * Datepicker lets you select dates.
  3 * @name DatePicker
  4 * @class DatePicker
  5 * @augments ch.Controls
  6 * @requires ch.Calendar
  7 * @see ch.Controls
  8 * @see ch.Calendar
  9 * @memberOf ch
 10 * @param {Object} [conf] Object with configuration properties.
 11 * @param {String} [conf.format] Sets the date format. By default is "DD/MM/YYYY".
 12 * @param {String} [conf.selected] Sets a date that should be selected by default. By default is the date of today.
 13 * @param {String} [conf.from] Set a maximum selectable date.
 14 * @param {String} [conf.to] Set a minimum selectable date.
 15 * @param {String} [conf.points] Points to be positioned. See Positioner component. By default is "ct cb".
 16 * @param {Array} [conf.monthsNames] By default is ["Enero", ... , "Diciembre"].
 17 * @param {Array} [conf.weekdays] By default is ["Dom", ... , "Sab"].
 18 * @param {Boolean} [conf.closable] Defines if floated component will be closed when a date is selected or not. By default it's "true".
 19 * @returns itself
 20 * @factorized
 21 * @exampleDescription Create a new datePicker.
 22 * @example
 23 * var widget = $(".example").datePicker();
 24 * @exampleDescription Create a new Date Picker with configuration.
 25 * @example
 26 * var widget = $(".example").datePicker({
 27 *	 "format": "MM/DD/YYYY",
 28 *	 "selected": "2011/12/25",
 29 *	 "from": "2010/12/25",
 30 *	 "to": "2012/12/25",
 31 *	 "monthsNames": ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"],
 32 *	 "weekdays": ["Su", "Mo", "Tu", "We", "Thu", "Fr", "Sa"]
 33 * });
 34 */
 35 
 36 ch.datePicker = function (conf) {
 37 
 38 	/**
 39 	* Reference to a internal component instance, saves all the information and configuration properties.
 40 	* @private
 41 	* @name ch.DatePicker#that
 42 	* @type object
 43 	*/
 44 	var that = this;
 45 
 46 	conf = ch.clon(conf);
 47 
 48 	// Configuration by default
 49 	conf.format = conf.format || "DD/MM/YYYY";
 50 	conf.points = conf.points || "ct cb";
 51 	conf.closable = ch.utils.hasOwn(conf, "closable") ? conf.closable : true;
 52 
 53 	that.conf = conf;
 54 
 55 /**
 56 *	Inheritance
 57 */
 58 
 59 	that = ch.controls.call(that);
 60 	that.parent = ch.clon(that);
 61 
 62 /**
 63 *	Private Members
 64 */
 65 
 66 /**
 67 *	Protected Members
 68 */
 69 
 70 	/**
 71 	* Pick a date in the Calendar and updates the input data.
 72 	* @protected
 73 	* @function
 74 	* @name ch.DatePicker#process
 75 	*/
 76 	that.process = function (event) {
 77 
 78 		// Day selection
 79 		if (event.target.nodeName !== "TD" || event.target.className.indexOf("ch-calendar-disabled") !== -1 || event.target.className.indexOf("ch-calendar-other") !== -1) {
 80 			return;
 81 		}
 82 
 83 		// Select the day and update input value with selected date
 84 
 85 		that.element.value = that.calendar.selectDay(event.target.innerHTML);
 86 
 87 		// Hide float
 88 		if (conf.closable) { that["float"].innerHide(); }
 89 
 90 		/**
 91 		* Callback function
 92 		* @public
 93 		* @name ch.DatePicker#select
 94 		* @event
 95 		*/
 96 		// Old callback system
 97 		that.callbacks("onSelect");
 98 		// New callback
 99 		that.trigger("select");
100 
101 	};
102 
103 
104 	/**
105 	* Reference to the Calendar component instance.
106 	* @protected
107 	* @type Object
108 	* @name ch.DatePicker#calendar
109 	*/
110 	that.calendar = $("<div>")
111 		// Add functionality for date selection
112 		.on("click", function (event) { that.process(event); })
113 		// Instance Calendar component
114 		.calendar({
115 			"format": conf.format,
116 			"from": conf.from,
117 			"to": conf.to,
118 			"selected": conf.selected,
119 			"monthsNames": conf.monthsNames,
120 			"weekdays": conf.weekdays
121 		});
122 
123 	/**
124 	* Reference to the Float component instanced.
125 	* @protected
126 	* @type Object
127 	* @name ch.DatePicker#float
128 	*/
129 	that["float"] = that.createFloat({
130 		"$element": $("<i role=\"button\" class=\"ch-datePicker-trigger ch-icon-calendar\"></i>").insertAfter(that.element),
131 		"content": that.calendar.element,
132 		"points": conf.points,
133 		"offset": "-1 8",
134 		"aria": {
135 			"role": "tooltip",
136 			"identifier": "aria-describedby"
137 		},
138 		"closeButton": false,
139 		"cone": true
140 	});
141 
142 /**
143 *  Public Members
144 */
145 
146 	/**
147 	* @borrows ch.Object#uid as ch.Menu#uid
148 	*/
149 
150 	/**
151 	* @borrows ch.Object#element as ch.Menu#element
152 	*/
153 
154 	/**
155 	* @borrows ch.Object#type as ch.Menu#type
156 	*/
157 
158 	/**
159 	* Triggers the innerShow method and returns the public scope to keep method chaining.
160 	* @public
161 	* @name ch.DatePicker#show
162 	* @function
163 	* @returns itself
164 	* @exampleDescription Following the first example, using <code>widget</code> as modal's instance controller:
165 	* @example
166 	* widget.show();
167 	*/
168 	that["public"].show = function () {
169 
170 		that["float"].innerShow();
171 
172 		return that["public"];
173 	};
174 
175 	/**
176 	* Triggers the innerHide method and returns the public scope to keep method chaining.
177 	* @public
178 	* @name ch.DatePicker#show
179 	* @function
180 	* @returns itself
181 	* @exampleDescription Following the first example, using <code>widget</code> as modal's instance controller:
182 	* @example
183 	* widget.hide();
184 	*/
185 	that["public"].hide = function () {
186 		that["float"].innerHide();
187 
188 		return that["public"];
189 	};
190 
191 	/**
192 	* Select a specific date or returns the selected date.
193 	* @public
194 	* @since 0.9
195 	* @name ch.DatePicker#select
196 	* @function
197 	* @param {string} "YYYY/MM/DD".
198 	* @return itself
199 	*/
200 	that["public"].select = function (date) {
201 		// Select the day and update input value with selected date
202 		// Setter
203 		if (date) {
204 			that.calendar.select(date);
205 			that.element.value = that.calendar.select();
206 
207 			return that["public"];
208 		}
209 
210 		// Getter
211 		return that.calendar.select();
212 	};
213 
214 	/**
215 	* Returns date of today
216 	* @public
217 	* @since 0.9
218 	* @name ch.DatePicker#today
219 	* @function
220 	* @return date
221 	*/
222 	that["public"].today = function () {
223 		return that.calendar.today();
224 	};
225 
226 	/**
227 	* Move to the next month or year. If it isn't specified, it will be moved to next month.
228 	* @public
229 	* @function
230 	* @name ch.DatePicker#next
231 	* @param {String} time A string that allows specify if it should move to next month or year.
232 	* @return itself
233 	* @default Next month
234 	*/
235 	that["public"].next = function (time) {
236 		that.calendar.next(time);
237 
238 		return that["public"];
239 	};
240 
241 	/**
242 	* Move to the previous month or year. If it isn't specified, it will be moved to previous month.
243 	* @public
244 	* @function
245 	* @name ch.DatePicker#prev
246 	* @param {String} time A string that allows specify if it should move to previous month or year.
247 	* @return itself
248 	* @default Previous month
249 	*/
250 	that["public"].prev = function (time) {
251 		that.calendar.prev(time);
252 
253 		return that["public"];
254 	};
255 
256 	/**
257 	* Reset the Date Picker to date of today
258 	* @public
259 	* @name ch.DatePicker#reset
260 	* @function
261 	* @return itself
262 	*/
263 	that["public"].reset = function () {
264 
265 		// Delete input value
266 		that.element.value = "";
267 
268 		that.calendar.reset();
269 
270 		return that["public"];
271 	};
272 
273 	/**
274 	* Set a minimum selectable date.
275 	* @public
276 	* @name ch.DatePicker#from
277 	* @function
278 	* @param {string} "YYYY/MM/DD".
279 	* @return itself
280 	*/
281 	that["public"].from = function (date) {
282 		that.calendar.from(date);
283 
284 		return that["public"];
285 	};
286 
287 	/**
288 	* Set a maximum selectable date.
289 	* @public
290 	* @name ch.DatePicker#to
291 	* @function
292 	* @param {string} "YYYY/MM/DD".
293 	* @return itself
294 	*/
295 	that["public"].to = function (date) {
296 		that.calendar.to(date);
297 
298 		return that["public"];
299 	};
300 
301 
302 /**
303 *	Default event delegation
304 */
305 
306 	// Change type of input to "text"
307 	that.element.type = "text";
308 
309 	// Change value of input if there are a selected date
310 	that.element.value = (conf.selected) ? that.calendar.select() : that.element.value;
311 
312 	// Add show behaivor to float's trigger.
313 	that["float"].$element.on("click", function (event) {
314 		that["float"].innerShow(event);
315 	});
316 
317 	/**
318 	* Triggers when the component is ready to use (Since 0.8.0).
319 	* @name ch.DatePicker#ready
320 	* @event
321 	* @public
322 	* @exampleDescription Following the first example, using <code>widget</code> as Date Picker's instance controller:
323 	* @example
324 	* widget.on("ready", function () {
325 	* 	this.show();
326 	* });
327 	*/
328 	setTimeout(function () { that.trigger("ready"); }, 50);
329 
330 	return that;
331 };
332 
333 ch.factory("datePicker");
