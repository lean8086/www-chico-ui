1 /**
  2 * Dropdown shows a list of options for navigation.
  3 * @name Dropdown
  4 * @class Dropdown
  5 * @augments ch.Navs
  6 * @requires ch.Positioner
  7 * @see ch.Navs
  8 * @see ch.Positioner
  9 * @see ch.Expando
 10 * @see ch.TabNavigator
 11 * @memberOf ch
 12 * @param {Object} [conf] Object with configuration properties.
 13 * @param {Boolean} [conf.open] Shows the dropdown open when component was loaded. By default, the value is false.
 14 * @param {Boolean} [conf.icon] Shows an arrow as icon. By default, the value is true.
 15 * @param {String} [conf.points] Sets the points where component will be positioned, specified by configuration or centered by default: "cm cm".
 16 * @param {Boolean} [conf.fx] Enable or disable UI effects. By default, the effects are disable.
 17 * @returns itself
 18 * @factorized
 19 * @exampleDescription Create a new dropdown without configuration.
 20 * @example
 21 * var widget = $(".example").dropdown();
 22 * @exampleDescription Create a new dropdown with configuration.
 23 * @example
 24 * var widget = $(".example").dropdown({
 25 *     "open": true,
 26 *     "icon": false,
 27 *     "points": "lt lt",
 28 *     "fx": true
 29 * });
 30 */
 31 
 32 ch.dropdown = function (conf) {
 33 
 34 	/**
 35 	* Reference to a internal component instance, saves all the information and configuration properties.
 36 	* @private
 37 	* @name ch.Dropdown#that
 38 	* @type object
 39 	*/
 40 	var that = this;
 41 
 42 	conf = ch.clon(conf);
 43 
 44 	conf.reposition = ch.utils.hasOwn(conf, "reposition") ? conf.reposition : true;
 45 
 46 	that.conf = conf;
 47 
 48 /**
 49 *	Inheritance
 50 */
 51 
 52 	that = ch.navs.call(that);
 53 	that.parent = ch.clon(that);
 54 
 55 /**
 56 *  Private Members
 57 */
 58 
 59 
 60 /**
 61 *  Protected Members
 62 */
 63 	/**
 64 	* The component's trigger.
 65 	* @protected
 66 	* @name ch.Dropdown#$trigger
 67 	* @type jQuery
 68 	*/
 69 	that.$trigger = (function () {
 70 
 71 		var $el = that.$trigger;
 72 
 73 		if (!that.$element.hasClass("ch-dropdown-skin")) {
 74 			$el.addClass("ch-btn-skin ch-btn-small");
 75 		}
 76 
 77 		return $el;
 78 
 79 	}());
 80 
 81 	/**
 82 	* The component's content.
 83 	* @protected
 84 	* @name ch.Dropdown#$content
 85 	* @type jQuery
 86 	*/
 87 	that.$content = (function () {
 88 
 89 		// jQuery Object
 90 		var $content = that.$content
 91 		// Prevent click on content (except links)
 92 			.bind("click", function(event) {
 93 				if ((event.target || event.srcElement).tagName === "A") {
 94 					that.innerHide();
 95 				}
 96 				event.stopPropagation();
 97 			})
 98 		// WAI-ARIA properties
 99 			.attr({ "role": "menu", "aria-hidden": "true" });
100 
101 		// WAI-ARIA for options into content
102 		$content.children("a").attr("role", "menuitem");
103 
104 		// Position
105 		that.position = ch.positioner({
106 			"element": $content,
107 			"context": that.$trigger,
108 			"points": (conf.points || "lt lb"),
109 			"offset": "0 -1",
110 			"reposition": conf.reposition
111 		});
112 
113 		return $content;
114 	}());
115 
116 
117 	/**
118 	* Dropdown options.
119 	* @protected
120 	* @jQuery Collection
121 	* @name ch.Dropdown#$options
122 	*/
123 	that.$options = that.$content.find("a");
124 
125 
126 	/**
127 	* Keyboard events object.
128 	* @protected
129 	* @Object
130 	* @name ch.Dropdown#shortcuts
131 	*/
132 	that.shortcuts = (function () {
133 		var selected,
134 			map = {},
135 			arrow,
136 			optionsLength = that.$options.length,
137 			selectOption = function (key, event) {
138 				// Validations
139 				if (!that.active) { return; }
140 
141 				// Prevent default behaivor
142 				that.prevent(event);
143 
144 				// Sets the arrow that user press
145 				arrow = key.type;
146 
147 				// Sets limits behaivor
148 				if (selected === (arrow === "down_arrow" ? optionsLength - 1 : 0)) { return; }
149 
150 				// Unselects current option
151 				that.$options[selected].blur();
152 
153 				if (arrow === "down_arrow") { selected += 1; } else { selected -= 1; }
154 
155 				// Selects new current option
156 				that.$options[selected].focus();
157 			};
158 
159 		return function () {
160 			// Keyboard support initialize
161 			selected = 0;
162 
163 			// Item selected by mouseover
164 			$.each(that.$options, function (i, e) {
165 				$(e).bind("mouseenter", function () {
166 					that.$options[selected = i].focus();
167 				});
168 			});
169 
170 			// Creates keyboard shortcuts map and binding events
171 			map["click " + ch.events.KEY.ESC] = function () { that.innerHide();Â };
172 			map[ch.events.KEY.UP_ARROW + " " +ch.events.KEY.DOWN_ARROW] = selectOption;
173 			ch.utils.document.on(map);
174 		}
175 	}());
176 
177 	/**
178 	* Shows component's content.
179 	* @protected
180 	* @function
181 	* @name ch.Dropdown#innerShow
182 	* @returns itself
183 	*/
184 	that.innerShow = function (event) {
185 
186 		// Stop propagation
187 		that.prevent(event);
188 
189 		// Z-index of content and updates aria values
190 		that.$content.css("z-index", ch.utils.zIndex += 1).attr("aria-hidden", "false");
191 
192 		// Z-index of trigger over content (secondary / skin dropdown)
193 		if (that.$element.hasClass("ch-dropdown-skin")) { that.$trigger.css("z-index", ch.utils.zIndex += 1); }
194 
195 		// Inheritance innerShow
196 		that.parent.innerShow(event);
197 
198 		// Refresh position
199 		that.position("refresh");
200 
201 		// Reset all dropdowns except itself
202 		$.each(ch.instances.dropdown, function (i, e) {
203 			if (e.uid !== that.uid) { e.hide(); }
204 		});
205 
206 		that.$options[0].focus();
207 
208 		return that;
209 	};
210 
211 	/**
212 	* Hides component's content.
213 	* @protected
214 	* @function
215 	* @name ch.Dropdown#innerHide
216 	* @returns itself
217 	*/
218 	that.innerHide = function (event) {
219 
220 		// Inheritance innerHide
221 		that.parent.innerHide(event);
222 
223 		// Updates aria values
224 		that.$content.attr("aria-hidden", "true");
225 
226 		return that;
227 	};
228 
229 /**
230 *  Public Members
231 */
232 
233 	/**
234 	* @borrows ch.Object#uid as ch.Menu#uid
235 	*/
236 
237 	/**
238 	* @borrows ch.Object#element as ch.Menu#element
239 	*/
240 
241 	/**
242 	* @borrows ch.Object#type as ch.Menu#type
243 	*/
244 
245 	/**
246 	* @borrows ch.Navs#show as ch.Dropdown#type
247 	*/
248 
249 	/**
250 	* @borrows ch.Navs#hide as ch.Dropdown#hide
251 	*/
252 
253 	/**
254 	* Positioning configuration.
255 	* @public
256 	* @name ch.Dropdown#position
257 	* @function
258 	*/
259 	that["public"].position = that.position;
260 
261 /**
262 *  Default event delegation
263 */
264 
265 	ch.utils.avoidTextSelection(that.$trigger);
266 
267 	// Inits keyboards support
268 	that.shortcuts();
269 
270 	/**
271 	* Triggers when the component is ready to use (Since 0.8.0).
272 	* @name ch.Dropdown#ready
273 	* @event
274 	* @public
275 	* @since 0.8.0
276 	* @exampleDescription Following the first example, using <code>widget</code> as dropdown's instance controller:
277 	* @example
278 	* widget.on("ready",function () {
279 	*	this.show();
280 	* });
281 	*/
282 	setTimeout(function(){ that.trigger("ready")}, 50);
283 
284 	return that;
285 };
286 
287 ch.factory("dropdown");
