  2 /**
  3 * Forms is a Controller of DOM's HTMLFormElement.
  4 * @name Form
  5 * @class Form
  6 * @augments ch.Controllers
  7 * @standalone
  8 * @memberOf ch
  9 * @param {Object} [conf] Object with configuration properties.
 10 * @param {Object} [conf.messages]
 11 * @see ch.Watchers
 12 * @returns itself
 13 * @example
 14 * // Create a new form with configuration.
 15 * var me = $(".example").form({
 16 * 	"messages": {
 17 * 		"required": "Error message for all required fields.",
 18 * 		"email": "Show this message on email format error."
 19 * 	}
 20 * });
 21 * @example
 22 * // Create a new form without configuration.
 23 * var me = $(".example").form();
 24 */
 25 
 26 ch.form = function(conf) {
 27 
 28 /**
 29 * Validation
 30 */
 31 	// Are there action and submit type?
 32 	if ( this.$element.find(":submit").length == 0 || this.$element.attr("action") == "" ){
 33 		alert("Form fatal error: The <input type=submit> is missing, or need to define a action attribute on the form tag.");
 34 		return;
 35 	};
 36 
 37 	// Is there form in map instances?
 38 	if ( ch.utils.hasOwn(ch.instances, "form") && ch.instances.form.length > 0 ){
 39 		for(var i = 0, j = ch.instances.form.length; i < j; i+=1){
 40 			if(ch.instances.form[i].element === this.element){
 41 				return {
 42 					exists: true,
 43 					object: ch.instances.form[i]
 44 				};
 45 			}
 46 		};
 47 	}
 48 
 49 	/**
 50 	* Reference to a internal component instance, saves all the information and configuration properties.
 51 	* @private
 52 	* @name ch.Form#that
 53 	* @type object
 54 	*/
 55 	var that = this;
 56 
 57 	conf = ch.clon(conf);
 58 	// Disable HTML5 browser-native validations
 59 	that.$element.attr("novalidate", "novalidate");
 60 	// Grab submit button
 61 	that.$submit = that.$element.find("input:submit");
 62 
 63 	that.conf = conf;
 64 
 65 /**
 66 *  Inheritance
 67 */
 68 
 69 	that = ch.controllers.call(that);
 70 	that.parent = ch.clon(that);
 71 	
 72 /**
 73 *  Private Members
 74 */
 75 
 76 	/**
 77 	* A Boolean property that indicates is exists errors in the form.
 78 	* @private
 79 	* @name ch.Form#status
 80 	* @type boolean
 81 	*/
 82 	var status = true;
 83 
 84 	/**
 85 	* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
 86 	*/
 87 	var validate = function(event){
 88 
 89 		/**
 90 		* Callback function
 91 		* @name ch.Form#beforeValidate
 92 		* @event
 93 		* @public
 94 		*/
 95 		that.callbacks("beforeValidate");
 96 		// new callback
 97 		that.trigger("beforeValidate");
 98 
 99 		// Status OK (with previous error)
100 		if ( !status ) {
101 			status = true;
102 		};
103 
104 		var i = 0, j = that.children.length, toFocus, childrenError = [];
105 
106 		// Shoot validations
107 		for (i; i < j; i+=1) {
108 			var child = that.children[i];
109 
110 			// Validate
111 			// Save children with errors
112 			if ( child.hasError() ) {
113 				childrenError.push(child);
114 			}
115 		};
116 
117 		// Is there's an error
118 		if (childrenError.length > 0) {
119 			status = false;
120 			// Issue UI-332: On validation must focus the first field with errors.
121 			// Doc: http://wiki.ml.com/display/ux/Mensajes+de+error
122 			if (childrenError[0].element.tagName === "DIV") {
123 				$(childrenError[0].element).find("input:first").focus();
124 			} else if (childrenError[0].element.type !== "hidden") {
125 				childrenError[0].element.focus();
126 			}
127 		} else {
128 			status = true;
129 		}
130 
131 		/**
132 		* Callback function
133 		* @name ch.Form#validate
134 		* @event
135 		* @public
136 		*/
137 		/**
138 		* Callback function
139 		* @name ch.Form#error
140 		* @event
141 		* @public
142 		*/
143 		if (status) {
144 			that.callbacks("onValidate");
145 			// new callback
146 			that.trigger("validate");
147 		} else {
148 			that.callbacks("onError");
149 			// new callback
150 			that.trigger("error");
151 		}
152 
153 		/**
154 		* Callback function
155 		* @name ch.Form#afterValidate
156 		* @event
157 		* @public
158 		*/
159 		that.callbacks("afterValidate");
160 		// new callback
161 		that.trigger("afterValidate");
162 
163 		return that;
164 	};
165 
166 	/**
167 	* This methods triggers the 'beforSubmit' callback, then will execute validate() method,
168 	* and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
169 	*/
170 	var submit = function(event) {
171 
172 		/**
173 		* Callback function
174 		* @name ch.Form#beforeSubmit
175 		* @event
176 		* @public
177 		*/
178 		that.callbacks("beforeSubmit");
179 		// new callback
180 		that.trigger("beforeSubmit");
181 
182 		// Execute all validations
183 		validate(event);
184 
185 		// If an error occurs prevent default actions
186 		if (!status) {
187 			that.prevent(event);
188 	        if (event) {
189 	            event.stopImmediatePropagation();
190 	        }
191 		}
192 
193 		// OLD CALLBACK SYSTEM!
194 		// Is there's no error but there's a onSubmit callback
195 		if ( status && ch.utils.hasOwn(conf, "onSubmit")) {
196 			// Avoid default actions
197 			that.prevent(event);
198 			// To execute defined onSubmit callback
199 			that.callbacks("onSubmit");
200 		}
201 
202 		/**
203 		* Callback function
204 		* @name ch.Form#submit
205 		* @event
206 		* @public
207 		*/
208 		// * New callback system *
209 		// Check inside $.data if there's a handler for ch-submit event
210 		// if something found there, avoid submit.
211 
212 		var formEvents = $(that["public"]).data("events");
213 		var isSubmitEventDefined = (formEvents && ch.utils.hasOwn(formEvents, "ch-submit"));
214 
215 		if (status && isSubmitEventDefined){
216 			// Avoid default actions
217 			that.prevent(event);
218 			// new callback
219 			that.trigger("submit");
220 		};
221 
222 		/**
223 		* Callback function
224 		* @name ch.Form#afterSubmit
225 		* @event
226 		* @public
227 		*/
228 		that.callbacks("afterSubmit");
229 		// new callback
230 		that.trigger("afterSubmit");
231 
232 		// Return that to chain methods
233 		return that;
234 	};
235 
236 	/**
237 	* Use this method to clear al validations.
238 	*/
239 	var clear = function(){
240 
241 		var i = 0, j = that.children.length;
242 		for(i; i < j; i += 1) {
243 			that.children[i].clear();
244 		}
245 
246 		status = true;
247 
248 		/**
249 		* Callback function
250 		* @name ch.Form#onClear
251 		* @event
252 		* @public
253 		*/
254 		that.callbacks("onClear");
255 		// new callback
256 		that.trigger("clear");
257 
258 		return that;
259 	};
260 
261 	/**
262 	* Use this method to reset the form's input elements.
263 	*/
264 	var reset = function(event){
265 		clear();
266 		that.element.reset(); // Reset html form native
267 
268 		/**
269 		* Callback function
270 		* @name ch.Form#onReset
271 		* @event
272 		* @public
273 		*/
274 		that.callbacks("onReset");
275 		// new callback
276 		that.trigger("reset");
277 
278 		return that;
279 	};

307 	* Watcher instances associated to this controller.
308 	* @public
309 	* @name ch.Form#children
310 	* @type collection
311 	*/
312 	that["public"].children = that.children;
313 
314 	/**
315 	* Collection of messages defined.
316 	* @public
317 	* @name ch.Form#messages
318 	* @type string
319 	*/
320 	that["public"].messages = conf.messages || {};
321 
322 	/**
323 	* Executes all children's validations, if finds a error will trigger 'onError' callback, if no error is found will trigger 'onValidate' callback, and allways trigger 'afterValidate' callback.
324 	* @public
325 	* @function
326 	* @name ch.Form#validate
327 	* @returns itself
328 	*/
329 	that["public"].validate = function() {
330 		validate();
331 
332 		return that["public"];
333 	};
334 
335 	/**
336 	* This methods triggers the 'beforSubmit' callback, then will execute validate() method, and if is defined triggers 'onSubmit' callback, at the end will trigger the 'afterSubmit' callback.
337 	* @public
338 	* @function
339 	* @name ch.Form#submit
340 	* @returns itself
341 	*/
342 	that["public"].submit = function() {
343 		submit();
344 
345 		return that["public"];
346 	};
347 
348 	/**
349 	* Return the status value.
350 	* @public
351 	* @function
352 	* @name ch.Form#getStatus
353 	* @returns itself
354 	*/
355 	that["public"].getStatus = function(){
356 		return status;
357 	};
358 
359 	/**
360 	* Use this method to clear al validations.
361 	* @public
362 	* @function
363 	* @name ch.Form#clear
364 	* @returns itself
365 	*/
366 	that["public"].clear = function() {
367 		clear();
368 
369 		return that["public"];
370 	};
371 
372 	/**
373 	* Use this method to clear al validations.
374 	* @public
375 	* @function
376 	* @name ch.Form#reset
377 	* @returns itself
378 	*/
379 	that["public"].reset = function() {
380 		reset();
381 
382 		return that["public"];
383 	};
384 
385 /**
386 *  Default event delegation
387 */
388 
389 	// patch exists because the components need a trigger
390 	if (ch.utils.hasOwn(conf, "onSubmit")) {
391 		that.$element.bind('submit', function(event){ that.prevent(event); });
392 		// Delete all click handlers asociated to submit button >NATAN: Why?
393 			//Because if you want to do something on submit, you need that the trigger (submit button)
394 			//don't have events associates. You can add funcionality on onSubmit callback
395 		that.$element.find(":submit").unbind('click');
396 	};
397 
398 	// Bind the submit
399 	that.$element.bind("submit", function(event){ submit(event) });
400 
401 	// Bind the reset
402 	that.$element.find(":reset, .resetForm").bind("click", function(event){ reset(event); });
403 
404 	/**
405 	* Triggers when the component is ready to use (Since 0.8.0).
406 	* @name ch.Form#ready
407 	* @event
408 	* @public
409 	* @since 0.8.0
410 	* @example
411 	* // Following the first example, using 'me' as form's instance controller:
412 	* me.on("ready",function () {
413 	*	this.reset();
414 	* });
415 	*/
416 	setTimeout(function(){ that.trigger("ready")}, 50);
417 
418 	return that;
419 };
